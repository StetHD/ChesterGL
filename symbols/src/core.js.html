188 	/**
189 	 * sets the current program, also sets the uniforms for that shader
190 	 * 
191 	 * @function
192 	 * @param {String} program
193 	 */
194 	ChesterGL.selectProgram = function (program) {
195 		var prog = this.programs[program];
196 		var gl = this.gl;
197 		if (program != this.currentProgram) {
198 			// console.log("selecting program " + program);
199 			this.currentProgram = program;
200 			gl.validateProgram(prog);
201 			gl.useProgram(prog);
202 			// enable attribs
203 			for (var attr in prog.attribs) {
204 				// console.log("  enabling attribute " + attr);
205 				gl.enableVertexAttribArray(prog.attribs[attr]);
206 			}
207 			// set the projection matrix
208 			gl.uniformMatrix4fv(prog.pMatrixUniform, false, this.pMatrix);
209 		}
210 		return prog;
211 	}

228 	/**
229 	 * tryies to init the graphics stuff:
230 	 * 1st attempt: webgl
231 	 * fallback: canvas
232 	 */
233 	ChesterGL.initGraphics = function (canvas) {
234 		try {
235 			this.canvas = canvas;
236 			if (this.webglMode) {
237 				this.gl = canvas.getContext("experimental-webgl");
238 			}
239 		} catch (e) {
240 			console.log("ERROR: " + e);
241 		}
242 		if (!this.gl) {
243 			// fallback to canvas API (uses an offscreen buffer)
244 			this.gl = canvas.getContext("2d");
245 			if (this.usesOffscreenBuffer) {
246 				this.offCanvas = document.createElement('canvas');
247 				this.offCanvas.width = canvas.width;
248 				this.offCanvas.height = canvas.height;
249 				this.offContext = this.offCanvas.getContext("2d");
250 				this.offContext.viewportWidth = canvas.width;
251 				this.offContext.viewportHeight = canvas.height;
252 				this['offContext'] = this.offContext;
253 				this.offContext['viewportWidth'] = this.offContext.viewportWidth;
254 				this.offContext['viewportHeight'] = this.offContext.viewportHeight;
255 			} else {
256 				this.offContext = this.gl;
257 			}
258 			if (!this.gl || !this.offContext) {
259 				throw "Error initializing graphic context!";
260 			}
261 			this.webglMode = false;
262 		}
263 		this['gl'] = this.gl;
264 		
265 		// get real width and height
266 		this.gl.viewportWidth = canvas.width;
267 		this.gl.viewportHeight = canvas.height;
268 		this.gl['viewportWidth'] = this.gl.viewportWidth;
269 		this.gl['viewportHeight'] = this.gl.viewportHeight;
270 	}

299 	/**
300 	 * init shaders (fetches data - in a sync way)
301 	 */
302 	ChesterGL.initShader = function (prefix, callback) {
303 		var gl = this.gl;
304 		var fsData = this.loadShader(prefix, "frag");
305 		var vsData = this.loadShader(prefix, "vert");
306 		
307 		var fs = gl.createShader(gl.FRAGMENT_SHADER);
308 		gl.shaderSource(fs, fsData);
309 		gl.compileShader(fs);
310 		
311 		if (!gl.getShaderParameter(fs, gl.COMPILE_STATUS)) {
312 			console.log("problem compiling fragment shader " + prefix + "(" + gl.getShaderInfoLog(fs) + "):\n" + fsData);
313 			return;
314 		}
315 		
316 		var vs = gl.createShader(gl.VERTEX_SHADER);
317 		gl.shaderSource(vs, vsData);
318 		gl.compileShader(vs);
319 		if (!gl.getShaderParameter(vs, gl.COMPILE_STATUS)) {
320 			console.log("problem compiling vertex shader " + prefix + "(" + gl.getShaderInfoLog(vs) + "):\n" + vsData);
321 			return;
322 		}
323 		
324 		
325 		var program = this.createShader(prefix, fs, vs);
326 		if (callback) callback(program);
327 		this.initShader1 = true;
328 	}

350 	/**
351 	 * actually creates the shader
352 	 */
353 	ChesterGL.createShader = function (prefix, fragmentData, vertexData) {
354 		var gl = this.gl;
355 		var program = gl.createProgram();
356 		gl.attachShader(program, fragmentData);
357 		gl.attachShader(program, vertexData);
358 		gl.linkProgram(program);
359 		if (!gl.getProgramParameter(program, gl.LINK_STATUS)) {
360 			console.log("problem linking shader");
361 		}
362 		console.log("creating shader " + prefix);
363 		this.programs[prefix] = program;
364 		return program;
365 	}

400 	/**
401 	 * loads and asset asynchronically
402 	 * 
403 	 * @param {string} type the type of asset being loaded, it could be "texture", "frameset", "audio"
404 	 * @param {string|Object} path the path for the asset
405 	 * @param {function(Object)=} callback the callback that will be executed as soon as the asset is loaded
406 	 */
407 	ChesterGL.loadAsset = function (type, path, callback) {
408 		var dataType = undefined;
409 		if (typeof(path) == 'object') {
410 			dataType = path.dataType;
411 			path = path.path;
412 		}
413 		
414 		if (!this.assets[type]) {
415 			this.assets[type] = {};
416 		}
417 		var assets = this.assets[type];
418 		if (!assets[path]) {
419 			// not in our records
420 			console.log("loading asset [" + type + "] " + path);
421 			assets[path] = {
422 				data: null,
423 				status: 'try',
424 				listeners: []
425 			}
426 			callback && assets[path].listeners.push(callback);
427 			this.loadAsset(type, {path: path, dataType: dataType});
428 		} else if (assets[path].status == 'loading') {
429 			// created but not yet loaded
430 			callback && assets[path].listeners.push(callback);
431 		} else if (assets[path].status == 'loaded') {
432 			// created and loaded, just call the callback
433 			callback && callback(assets[path].data);
434 		} else if (assets[path].status == 'try') {
435 			assets[path].status = 'loading';
436 			$.ajax({
437 				url: path,
438 				dataType: dataType,
439 				beforeSend: function (xhr) {
440 					// only binary data, please
441 					xhr.overrideMimeType('text/plain; charset=x-user-defined');
442 				},
443 				success: function (data, textStatus) {
444 					if (textStatus == "success") {
445 						ChesterGL.assetsHandlers[type](path, data, function (handled) {
446 							if (!handled) {
447 								assets[path].status = 'try';
448 								console.log("fetching " + path + " - again");
449 								ChesterGL.loadAsset(type, {path: path, dataType: dataType});
450 							} else {
451 								assets[path].status = 'loaded';
452 								// call all listeners
453 								var l;
454 								while (l = assets[path].listeners.shift()) { l(assets[path].data); }
455 								ChesterGL.assetsLoaded(type);
456 								ChesterGL.assetsLoaded('all');
457 							}
458 						});
459 					} else {
460 						console.log("Error loading asset " + path);
461 					}
462 				}
463 			});
464 		}
465 	}

467 	/**
468 	 * adds a listener for when all assets are loaded
469 	 * 
470 	 * @param {string} type You can query for all types if you pass "all" as type
471 	 * @param {function()=} callback The callback to be executed when all assets of that type are loaded
472 	 */
473 	ChesterGL.assetsLoaded = function (type, callback) {
474 		var listeners = this.assetsLoadedListeners[type];
475 		if (!listeners) {
476 			this.assetsLoadedListeners[type] = [];
477 			listeners = this.assetsLoadedListeners[type];
478 		}
479 		if (callback) {
480 			listeners.push(callback);
481 		}
482 		var allLoaded = true;
483 		if (type == 'all') {
484 			for (var t in this.assets) {
485 				var assets = this.assets[t];
486 				for (var i in assets) {
487 					if (assets[i].status != 'loaded') {
488 						allLoaded = false;
489 						break;
490 					}
491 				}
492 				if (!allLoaded) {
493 					break;
494 				}
495 			}
496 		} else {
497 			var assets = this.assets[type];
498 			for (var i in assets) {
499 				if (assets[i].status != 'loaded') {
500 					allLoaded = false;
501 					break;
502 				}
503 			}
504 		}
505 		if (allLoaded) {
506 			var l;
507 			while (l = listeners.shift()) { l(); }
508 		}
509 	}

518 	/**
519 	 * handles a loaded texture - should only be called on a webGL mode
520 	 * @return {boolean}
521 	 */
522 	ChesterGL.prepareWebGLTexture = function (texture) {
523 		var gl = this.gl;
524 		var result = true;
525 		
526 		try {
527 			var error = 0;
528 			gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);
529 			gl.bindTexture(gl.TEXTURE_2D, texture.tex);
530 			gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, texture);
531 			error = gl.getError();
532 			if (error != 0) {
533 				console.log("gl error " + error);
534 				result = false;
535 			}
536 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
537 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
538 			gl.bindTexture(gl.TEXTURE_2D, null);
539 		} catch (e) {
540 			console.log("got some error: " + e);
541 			result = false;
542 		}
543 		return result;
544 	}

548 	 * 
549 	 * @param {string} path
550 	 * @param {string} data
551 	 * @param {function(boolean)=} callback
552 	 */
553 	ChesterGL.defaultTextureHandler = function (path, data, callback) {
554 		var imgtype = (/[.]/.exec(path)) ? /[^.]+$/.exec(path) : undefined;
555 		var img = new Image();
556 		var rawData = base64.encode(data);
557 		/** @ignore */
558 		img.onload = function () {
559 			if (ChesterGL.webglMode) {
560 				img.tex = ChesterGL.gl.createTexture();
561 			}
562 			var texture = ChesterGL.assets['texture'][path];
563 			texture.data = img;
564 			var result = true;
565 			if (ChesterGL.webglMode) {
566 				result = ChesterGL.prepareWebGLTexture(img);
567 			}
568 			callback && callback(result);
569 		}
570 		img.src = "data:image/" + imgtype + ";base64," + rawData;
571 	}

573 	/**
574 	 * setups the perspective (2d or 3d)
575 	 */
576 	ChesterGL.setupPerspective = function () {
577 		var gl = this.gl;
578 		
579 		// quick bail if we're not on a webgl rendering mode
580 		if (!this.webglMode) {
581 			return;
582 		}
583 		
584 		gl.clearColor(0.0, 0.0, 0.0, 1.0);
585 		gl.clearDepth(1.0);
586 		
587 		var width = gl.viewportWidth;
588 		var height = gl.viewportHeight;
589 		gl.viewport(0, 0, width, height);
590 		
591 		this.pMatrix = mat4.create();
592 		
593 		if (this.projection == "2d") {
594 			// 2d projection
595 			console.log("setting up 2d projection (" + width + "," + height + ")");
596 			mat4.ortho(0, width, 0, height, -1024, 1024, this.pMatrix);
597 		} else if (this.projection == "3d") {
598 			// 3d projection
599 			console.log("setting up 3d projection (" + width + "," + height + ")");
600 			// var f_aspect = (1.7320508075688776 / (width / height));
601 			var zeye   = height / 1.1566;
602 			var matA   = mat4.perspective(60, width / height, 0.5, 1500.0);
603 			var eye    = vec3.create([width/2, height/2, zeye]);
604 			var center = vec3.create([width/2, height/2, 0]);
605 			var up     = vec3.create([0, 1, 0]);
606 			var matB = mat4.lookAt(eye, center, up);
607 			mat4.multiply(matA, matB, this.pMatrix);
608 		} else {
609 			throw "Invalid projection: " + this.projection;
610 		}
611 	}

613 	/**
614 	 * main draw function, will call the root block
615 	 */
616 	ChesterGL.drawScene = function () {		
617 		if (this.webglMode) {
618 			var gl = this.gl;
619 			gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
620 
621 			// global blending options
622 			gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
623 			gl.enable(gl.BLEND);
624 		} else {
625 			var gl = this.offContext;
626 			gl.setTransform(1, 0, 0, 1, 0, 0);
627 			gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
628 		}
629 		
630 		// start mayhem
631 		if (this.runningScene) {
632 			this.runningScene.visit();
633 		}
634 		
635 		if (!this.webglMode) {
636 			// copy back the off context (if we use one)
637 			if (this.usesOffscreenBuffer) {
638 				this.gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
639 				this.gl.drawImage(this.offCanvas, 0, 0);
640 			}
641 		}
642 		
643 		// for actions and other stuff
644 		var current = new Date; // milliseconds
645 		this.delta = current - this.lastTime;
646 		if (this.delta > 150) {
647 			this.delta = 30; // set a lower delta when debugging
648 		}
649 		this.lastTime = current;	
650 	}
