 36 vec2.create = function (vec) {
 37 	var dest = new Float32Array(2);
 38 	
 39 	if (vec) {
 40 		dest[0] = vec[0];
 41 		dest[1] = vec[1];
 42 	}
 43

 53 /**
 54  * @ignore
 55  * @param {Event} event
 56  * @return {vec3}
 57  */
 58 HTMLCanvasElement.prototype.relativePosition = function (event) {
 59 	var pt = HTMLCanvasElement._canvas_tmp_mouse;
 60 	pt[0] = 0; pt[1] = 0;
 61 	if (event.x !== undefined && event.y !== undefined) {
 62 		pt[0] = event.x;
 63 		pt[1] = event.y;
 64 	} else {
 65 		pt[0] = event.clientX + document.body.scrollLeft + document.documentElement.scrollLeft;
 66 		pt[1] = event.clientY + document.body.scrollTop + document.documentElement.scrollTop;
 67 	}
 68 	pt[0] -= this.offsetLeft;
 69 	pt[1]  = this.height - (pt[1] - this.offsetTop);
 70 	return pt;
 71 };

265 	/**
266 	 * sets the current program, also sets the uniforms for that shader
267 	 * 
268 	 * @param {string} program
269 	 * @return {Object}
270 	 */
271 	ChesterGL.selectProgram = function (program) {
272 		var prog = this.programs[program];
273 		var gl = this.gl;
274 		if (program != this.currentProgram) {
275 			console.log("selecting program " + program);
276 			this.currentProgram = program;
277 			gl.validateProgram(prog);
278 			gl.useProgram(prog);
279 			// enable attribs
280 			for (var attr in prog.attribs) {
281 				// console.log("  enabling attribute " + attr);
282 				gl.enableVertexAttribArray(prog.attribs[attr]);
283 			}
284 		}
285 		return prog;
286 	}

306 	/**
307 	 * tryies to init the graphics stuff:
308 	 * 1st attempt: webgl
309 	 * fallback: canvas
310 	 * 
311 	 * @param {Element} canvas
312 	 */
313 	ChesterGL.initGraphics = function (canvas) {
314 		try {
315 			this.canvas = canvas;
316 			if (this.webglMode) {
317 				this.gl = canvas.getContext("experimental-webgl", {alpha: false, antialias: false});
318 				if (this.gl && window['WebGLDebugUtils']) {
319 					console.log("installing debug context");
320 					this.gl = WebGLDebugUtils.makeDebugContext(this.gl, throwOnGLError);
321 				}
322 			}
323 		} catch (e) {
324 			console.log("ERROR: " + e);
325 		}
326 		if (!this.gl) {
327 			// fallback to canvas API (uses an offscreen buffer)
328 			this.gl = canvas.getContext("2d");
329 			if (this.usesOffscreenBuffer) {
330 				this.offCanvas = document.createElement('canvas');
331 				this.offCanvas.width = canvas.width;
332 				this.offCanvas.height = canvas.height;
333 				this.offContext = this.offCanvas.getContext("2d");
334 				this.offContext.viewportWidth = canvas.width;
335 				this.offContext.viewportHeight = canvas.height;
336 				this.exportProperty(this, 'offContext', this.offContext);
337 				this.exportProperty(this.offContext, 'viewportWidth', this.offContext.viewportWidth);
338 				this.exportProperty(this.offContext, 'viewportHeight', this.offContext.viewportHeight);
339 			} else {
340 				this.offContext = this.gl;
341 			}
342 			if (!this.gl || !this.offContext) {
343 				throw "Error initializing graphic context!";
344 			}
345 			this.webglMode = false;
346 		}
347 		this.exportProperty(this, 'gl', this.gl);
348 		// first resize of the canvas
349 		this.canvasResized();
350 		
351 		// install touch handler
352 		this.installMouseHandlers();
353 	}

397 	/**
398 	 * init shaders (fetches data - in a sync way)
399 	 * @param {string} prefix
400 	 * @param {function(WebGLProgram)} callback
401 	 */
402 	ChesterGL.initShader = function (prefix, callback) {
403 		var gl = this.gl;
404 		var fsData = this.loadShader(prefix, "frag");
405 		var vsData = this.loadShader(prefix, "vert");
406 		
407 		var fs = gl.createShader(gl.FRAGMENT_SHADER);
408 		gl.shaderSource(fs, fsData);
409 		gl.compileShader(fs);
410 		
411 		if (!gl.getShaderParameter(fs, gl.COMPILE_STATUS)) {
412 			console.log("problem compiling fragment shader " + prefix + "(" + gl.getShaderInfoLog(fs) + "):\n" + fsData);
413 			return;
414 		}
415 		
416 		var vs = gl.createShader(gl.VERTEX_SHADER);
417 		gl.shaderSource(vs, vsData);
418 		gl.compileShader(vs);
419 		if (!gl.getShaderParameter(vs, gl.COMPILE_STATUS)) {
420 			console.log("problem compiling vertex shader " + prefix + "(" + gl.getShaderInfoLog(vs) + "):\n" + vsData);
421 			return;
422 		}
423 		
424 		var program = this.createShader(prefix, fs, vs);
425 		if (callback) callback(program);
426 	}

449 	/**
450 	 * actually creates the shader
451 	 * @param {string} prefix
452 	 * @param {(string|Object)} fragmentData
453 	 * @param {(string|Object)} vertexData
454 	 * @return {(WebGLProgram|null)}
455 	 */
456 	ChesterGL.createShader = function (prefix, fragmentData, vertexData) {
457 		var gl = this.gl;
458 		var program = gl.createProgram();
459 		gl.attachShader(program, fragmentData);
460 		gl.attachShader(program, vertexData);
461 		gl.linkProgram(program);
462 		if (!gl.getProgramParameter(program, gl.LINK_STATUS)) {
463 			console.log("problem linking shader");
464 		}
465 		// console.log("creating shader " + prefix);
466 		this.programs[prefix] = program;
467 		return program;
468 	}

513 	/**
514 	 * loads and asset using the registered method to download it
515 	 * 
516 	 * @param {string} type the type of asset being loaded, it could be "texture", "frameset", "audio"
517 	 * @param {string|Object} assetPath the path for the asset
518 	 * @param {function(Object)=} cb the callback that will be executed as soon as the asset is loaded
519 	 */
520 	ChesterGL.loadAsset = function (type, assetPath, cb) {
521 		var dataType_ = undefined;
522 		if (typeof(assetPath) == 'object') {
523 			dataType_ = assetPath.dataType;
524 			assetPath = assetPath.path;
525 		}
526 		
527 		if (!this.assets[type]) {
528 			this.assets[type] = {};
529 		}
530 		var assets = this.assets[type];
531 		if (!assets[assetPath]) {
532 			// not in our records
533 			assets[assetPath] = {
534 				data: null,
535 				status: 'try',
536 				listeners: []
537 			}
538 			cb && assets[assetPath].listeners.push(cb);
539 			this.loadAsset(type, {path: assetPath, dataType: dataType_});
540 		} else if (assets[assetPath].status == 'loading') {
541 			// created but not yet loaded
542 			cb && assets[assetPath].listeners.push(cb);
543 		} else if (assets[assetPath].status == 'loaded') {
544 			// created and loaded, just call the callback
545 			cb && cb(assets[assetPath].data);
546 		} else if (assets[assetPath].status == 'try') {
547 			assets[assetPath].status = 'loading';
548 			if (this.assetsLoaders[type])
549 				this.assetsLoaders[type](type, {url: assetPath, dataType: dataType_});
550 			else
551 				this.assetsLoaders['default'](type, {url: assetPath, dataType: dataType_});
552 			cb && assets[assetPath].listeners.push(cb);
553 		}
554 	}

556 	/**
557 	 * adds a listener for when all assets are loaded
558 	 * 
559 	 * @param {string} type You can query for all types if you pass "all" as type
560 	 * @param {function()=} callback The callback to be executed when all assets of that type are loaded
561 	 */
562 	ChesterGL.assetsLoaded = function (type, callback) {
563 		var listeners = this.assetsLoadedListeners[type];
564 		if (!listeners) {
565 			this.assetsLoadedListeners[type] = [];
566 			listeners = this.assetsLoadedListeners[type];
567 		}
568 		if (callback) {
569 			listeners.push(callback);
570 		}
571 		var allLoaded = true;
572 		if (type == 'all') {
573 			for (var t in this.assets) {
574 				var assets = this.assets[t];
575 				for (var i in assets) {
576 					if (assets[i].status != 'loaded') {
577 						allLoaded = false;
578 						break;
579 					}
580 				}
581 				if (!allLoaded) {
582 					break;
583 				}
584 			}
585 		} else {
586 			var assets = this.assets[type];
587 			for (var i in assets) {
588 				if (assets[i].status != 'loaded') {
589 					allLoaded = false;
590 					break;
591 				}
592 			}
593 		}
594 		if (allLoaded) {
595 			var l;
596 			while (l = listeners.shift()) { l(); }
597 		}
598 	}

610 	/**
611 	 * handles a loaded texture - should only be called on a webGL mode
612 	 * @param {(Object|HTMLImageElement)} texture
613 	 * @return {boolean}
614 	 */
615 	ChesterGL.prepareWebGLTexture = function (texture) {
616 		var gl = this.gl;
617 		var result = true;
618 		
619 		try {
620 			var error = 0;
621 			gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);
622 			gl.activeTexture(gl.TEXTURE0);
623 			gl.bindTexture(gl.TEXTURE_2D, texture.tex);
624 			gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, texture);
625 			error = gl.getError();
626 			if (error != 0) {
627 				console.log("gl error " + error);
628 				result = false;
629 			}
630 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
631 			gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
632 			gl.bindTexture(gl.TEXTURE_2D, null);
633 		} catch (e) {
634 			console.log("got some error: " + e);
635 			result = false;
636 		}
637 		return result;
638 	}

641 	 * The default texture handler
642 	 * 
643 	 * @param {string} path
644 	 * @param {Object} img
645 	 * @return {boolean}
646 	 */
647 	ChesterGL.defaultTextureHandler = function (path, img) {
648 		if (ChesterGL.webglMode) {
649 			img.tex = ChesterGL.gl.createTexture();
650 		}
651 		var texture = ChesterGL.assets['texture'][path];
652 		texture.data = img;
653 		if (ChesterGL.webglMode) {
654 			return ChesterGL.prepareWebGLTexture(img);
655 		}
656 		return true;
657 	}

659 	/**
660 	 * @param {string} type
661 	 * @param {Object.<string,string>} params
662 	 */
663 	ChesterGL.defaultTextureLoader = function (type, params) {
664 		var img = new Image();
665 		var path = params.url;
666 		img.addEventListener("load", function () {
667 			var texture = ChesterGL.assets['texture'][path];
668 			if (ChesterGL.assetsHandlers[type](path, img)) {
669 				// call all listeners
670 				texture.status = 'loaded';
671 				var l;
672 				while (l = texture.listeners.shift()) { l(texture.data); }
673 				// test for assets loaded
674 				ChesterGL.assetsLoaded(type);
675 				ChesterGL.assetsLoaded('all');
676 			} else {
677 				// requeue
678 				texture.status = 'try';
679 				ChesterGL.loadAsset(type, path);
680 			}
681 		}, false);
682 		img.src = path;
683 	}

689 	ChesterGL.defaultAssetLoader = function (type, params) {
690 		var path = params.url;
691 		$.ajax({
692 			url: path,
693 			dataType: params.dataType,
694 			success: function (data, textStatus) {
695 				var asset = ChesterGL.assets[type][path];
696 				if (textStatus == "success") {
697 					if (ChesterGL.assetsHandlers[type](path, data)) {
698 						asset.status = 'loaded';
699 						// call all listeners
700 						var l;
701 						while (l = asset.listeners.shift()) { l(asset.data); }
702 						// test for assets loaded
703 						ChesterGL.assetsLoaded(type);
704 						ChesterGL.assetsLoaded('all');
705 					} else {
706 						// requeue
707 						asset.status = 'try';
708 						ChesterGL.loadAsset(type, path);
709 					}
710 				} else {
711 					console.log("Error loading asset " + path);
712 				}
713 			}
714 		})
715 	}

717 	/**
718 	 * setups the perspective (2d or 3d)
719 	 */
720 	ChesterGL.setupPerspective = function () {
721 		var gl = this.gl;
722 		
723 		// quick bail if we're not on a webgl rendering mode
724 		if (!this.webglMode) {
725 			return;
726 		}
727 		
728 		gl.clearColor(0.0, 0.0, 0.0, 1.0);
729 		// gl.clearDepth(1.0);
730 		
731 		// global blending options
732 		gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
733 		gl.enable(gl.BLEND);
734 		// disable depth test
735 		gl.disable(gl.DEPTH_TEST)
736 		
737 		var width = gl.viewportWidth;
738 		var height = gl.viewportHeight;
739 		gl.viewport(0, 0, width, height);
740 		
741 		this.pMatrix = mat4.create();
742 		
743 		if (this.projection == "2d") {
744 			// 2d projection
745 			console.log("setting up 2d projection (" + width + "," + height + ")");
746 			mat4.ortho(0, width, 0, height, -1024, 1024, this.pMatrix);
747 		} else if (this.projection == "3d") {
748 			// 3d projection
749 			console.log("setting up 3d projection (" + width + "," + height + ")");
750 			// var f_aspect = (1.7320508075688776 / (width / height));
751 			var zeye   = height / 1.1566;
752 			var matA   = mat4.perspective(60, width / height, 0.5, 1500.0);
753 			var eye    = [width/2, height/2, zeye];
754 			var center = [width/2, height/2, 0];
755 			var up     = [0, 1, 0];
756 			var matB = mat4.lookAt(eye, center, up);
757 			mat4.multiply(matA, matB, this.pMatrix);
758 		} else {
759 			throw "Invalid projection: " + this.projection;
760 		}
761 	}

772 	/**
773 	 * main draw function, will call the root block
774 	 */
775 	ChesterGL.drawScene = function () {
776 		var gl = undefined;
777 		if (this.webglMode) {
778 			gl = this.gl;
779 			gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
780 		} else {
781 			gl = this.offContext;
782 			gl.setTransform(1, 0, 0, 1, 0, 0);
783 			gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
784 		}
785 		
786 		// start mayhem
787 		if (this.runningScene) {
788 			this.runningScene.visit();
789 		}
790 		
791 		if (!this.webglMode) {
792 			// copy back the off context (if we use one)
793 			if (this.usesOffscreenBuffer) {
794 				gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
795 				gl.drawImage(this.offCanvas, 0, 0);
796 			}
797 		}
798 		// for actions and other stuff
799 		var current = Date.now(); // milliseconds
800 		this.delta = current - this.lastTime;
801 		this.lastTime = current;	
802 	}

832 		var now = Date.now();
833 		this.elapsed_ += this.delta;
834 		this.frames_ ++;
835 		if (now - this.lastDebugSecond_ > 1000) {
836 			var avg = (this.elapsed_ / this.frames_);
837 			this.sumAvg += avg;
838 			this.sampledAvg ++;
839 			if (this.debugSpan) {
840 				this.debugSpan.textContent = avg.toFixed(2);
841 			}
842 			// track how well we're performing - every 5 seconds
843 			if (ChesterGL.runningScene && this.useGoogleAnalytics && this.sampledAvg > 5) {
844 				_gaq.push([
845 					'_trackEvent',
846 					'ChesterGL',
847 					// Let us know if this is WebGL or canvas mode
848 					'renderTime-' + (this.webglMode),
849 					ChesterGL.runningScene.title,
850 					Math.floor(this.sumAvg/this.sampledAvg)
851 				]);
852 				this.sumAvg = this.sampledAvg = 0;
853 			}
854 			this.elapsed_ = this.frames_ = 0;
855 			this.lastDebugSecond_ = now;
856 		}
