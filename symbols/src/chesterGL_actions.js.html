 31 /**
 32  * @constructor
 33  * @param {number} totalTime in milliseconds
 34  * @param {chesterGL.Block=} block The target block
 35  */
 36 chesterGL.Action = function (totalTime, block) {
 37 	this.totalTime = totalTime;
 38 	this.block = block;
 39 	this.elapsed = 0;
 40 };

 84 /**
 85  * This is the default update function (does nothing)
 86  * @param {number} delta
 87  * @ignore
 88  */
 89 chesterGL.Action.prototype.update = function (delta) {
 90 	this.elapsed += delta;
 91 	if (this.totalTime > 0 && this.elapsed >= this.totalTime) {
 92 		this.stop();
 93 	}
 94 };

 96 /**
 97  * will be called the first time - usually overriden by subclasses
 98  * @ignore
 99  */
100 chesterGL.Action.prototype.begin = function () {
101 	this.running = true;
102 };

104 /**
105  * will be called at the end of the function (or to stop it)
106  */
107 chesterGL.Action.prototype.stop = function () {
108 	this.finished = true;
109 	this.running = false;
110 };

113  * reset - prepare the action in order to use it again

117 	this.running = false;
118 	this.finished = false;
119 	this.elapsed = 0;

122 /**
123  * @constructor
124  * @param {Array|Float32Array} delta The final position (the initial position is the current one of the block)
125  * @param {number} totalTime The total time in seconds that this action should take
126  * @param {boolean=} relative whether or not the movement is relative (defaults: true)
127  * @param {chesterGL.Block=} block The block that will execute this action
128  * @extends {chesterGL.Action}
129  */
130 chesterGL.MoveAction = function (delta, totalTime, relative, block) {
131 	chesterGL.Action.call(this, totalTime, block);
132 	this.delta = goog.vec.Vec3.createFloat32FromArray(delta);
133 	if (relative !== undefined) {
134 		this.isRelative = (relative === true);
135 	} else {
136 		this.isRelative = true;
137 	}
138 	this.finalPosition = goog.vec.Vec3.createFloat32();
139 };
140 goog.inherits(chesterGL.MoveAction, chesterGL.Action);
141 
142 /**
143  * @type {?goog.vec.Vec3.Type}
144  */
145 chesterGL.MoveAction.prototype.delta = null;

150 chesterGL.MoveAction.prototype.finalPosition = null;
151 
152 /**
153  * @type {boolean}
154  */
155 chesterGL.MoveAction.prototype.isRelative = true;
156 
157 /**
158  * @type {?goog.vec.Vec3.Type}

160 chesterGL.MoveAction.prototype.startPosition = null;

162 /**
163  * @type {goog.vec.Vec3.Type}
164  * @ignore
165  */
166 chesterGL.MoveAction.__tmp_pos = goog.vec.Vec3.createFloat32();
167 
168 /**
169  * @param {number} delta miliseconds from last time we updated
170  * @ignore
171  */
172 chesterGL.MoveAction.prototype.update = function (delta) {
173 	chesterGL.Action.prototype.update.call(this, delta);
174 	var block = this.block;
175 	var t = Math.min(1, this.elapsed / this.totalTime);
176 	// console.log("t: " + t + "\t(" + dx + ")");
177 	goog.vec.Vec3.lerp(this.startPosition, this.finalPosition, t, chesterGL.MoveAction.__tmp_pos);
178 	block.setPosition(chesterGL.MoveAction.__tmp_pos);
179 };
180 
181 /**
182  * just set the initial position
183  * @ignore
184  */
185 chesterGL.MoveAction.prototype.begin = function () {
186 	chesterGL.Action.prototype.begin.call(this);
187 	if (!this.block) {
188 		throw "invalid move action! - no block";
189 	}
190 	if (this.isRelative) {
191 		goog.vec.Vec3.add(this.delta, this.block.position, this.finalPosition);
192 	} else {
193 		this.finalPosition = this.delta;
194 	}
195 	this.startPosition = this.block.position;
196 };
197 
198 /**
199  * @ignore
200  */
201 chesterGL.MoveAction.prototype.stop = function () {
202 	chesterGL.Action.prototype.stop.call(this);
203 	if (this.elapsed >= this.totalTime) {
204 		this.block.setPosition(this.finalPosition);
205 	}
206 };
207 
208 /**
209  * Return a new action with the reverse move action
210  * @return {chesterGL.MoveAction}
211  */
212 chesterGL.MoveAction.prototype.reverse = function () {
213 	if (!this.isRelative) {
214 		throw "This only works on relative movements";
215 	}
216 	var revDelta = [];
217 	goog.vec.Vec3.negate(this.delta, revDelta);
218 	return new chesterGL.MoveAction(revDelta, this.totalTime, true);
219 };

221 /**
222  * A simple action that will execute the callback, useful for
223  * sequences, e.g.: move, then execute some callback.
224  * @constructor
225  * @extends {chesterGL.Action}
226  * @param {function (Object=)} callback the callback to be executed
227  * @param {Object=} arg the object to be passed as argument to the
228  * callback.
229  * @example
230  * // move 100 points up in 0.5 seconds (500 milliseconds)
231  * var move = new chesterGL.MoveAction([0, 100, 0], 250);
232  * var remove = new chesterGL.CallbackAction(function () {
233  *   this.remove();
234  * }, someBlock);
235  */
236 chesterGL.CallbackAction = function (callback, arg) {
237 	this.callback = callback;
238 	this.arg = arg;
239 	chesterGL.Action.call(this, 1);
240 };
241 goog.inherits(chesterGL.CallbackAction, chesterGL.Action);

255 chesterGL.CallbackAction.prototype.update = function (delta) {

260 /**
261  * @constructor
262  * @param {chesterGL.Action} action1 the first action
263  * @param {chesterGL.Action} action2 the second action
264  * @extends {chesterGL.Action}
265  * @example
266  * var a1 = new chesterGL.MoveAction([100, 100, 0], 5000);
267  * var a2 = a1.reverse();
268  * var seq = new chesterGL.SequenceAction(a1, a2);
269  * block.runAction(seq);
270  */
271 chesterGL.SequenceAction = function (action1, action2) {
272 	var totalTime = (action1.totalTime + action2.totalTime);
273 	chesterGL.Action.call(this, totalTime);
274 	this.actions = [action1, action2];
275 };
276 goog.inherits(chesterGL.SequenceAction, chesterGL.Action);

278 /**
279  * @param {...chesterGL.Action} actions The list of actions to use in order to create the sequence
280  * @return {chesterGL.Action}
281  */
282 chesterGL.SequenceAction.createSequence = function (actions) {
283 	if (arguments.length === 0) {
284 		throw "Needs at least one action to create a sequence!";
285 	}
286 	var prev = arguments[0];
287 	for (var i=1; i < arguments.length; i++) {
288 		prev = new chesterGL.SequenceAction(prev, arguments[i]);
289 	}
290 	return prev;
291 };

312 /**
313  * just mark the split time (the duration of the first action)
314  * @ignore
315  */
316 chesterGL.SequenceAction.prototype.begin = function () {
317 	chesterGL.Action.prototype.begin.call(this);
318 	this.splitTime = this.actions[0].totalTime;
319 	this.block.runAction(this.actions[0]);
320 	// console.log("[seq begin] setting split time: " + this.splitTime);
321 };
322 
323 /**
324  * resets the sequence action (will also reset and unschedule its internal actions)
325  */
326 chesterGL.SequenceAction.prototype.reset = function () {
327 	chesterGL.Action.prototype.reset.call(this);
328 	this.currentAction = 0;
329 	this.actions[0].reset();
330 	this.actions[1].reset();
331 	chesterGL.ActionManager.unscheduleAction(this.actions[0].actionId);
332 	chesterGL.ActionManager.unscheduleAction(this.actions[1].actionId);
333 };
334 
335 /**
336  * propagate update to the corresponding action
337  * @ignore
338  */
339 chesterGL.SequenceAction.prototype.update = function (delta) {
340 	chesterGL.Action.prototype.update.call(this, delta);
341 	if (this.currentAction === 0 && this.elapsed >= this.splitTime) {
342 		// console.log("switching actions (" + this.actions[1].totalTime + "," + this.actions[1].elapsed + ")");
343 		this.actions[0].stop();
344 		this.currentAction = 1;
345 		this.block.runAction(this.actions[1]);
346 	}
347 };

349 /**
350  * @constructor
351  * @param {chesterGL.Action} action

355 chesterGL.RepeatAction = function (action, maxTimes) {
356 	this.maxTimes = maxTimes || 1;

392 /**
393  * @ignore
394  * @param {number} delta
395  */
396 chesterGL.RepeatAction.prototype.update = function (delta) {
397 	chesterGL.Action.prototype.update.call(this, delta);
398 	if (this.finished) {
399 		// console.log("repeat finished");
400 		if (this.maxTimes < 0 || this.times < this.maxTimes) {
401 			// console.log("repeating action");
402 			this.times++;
403 			this.reset();
404 			chesterGL.ActionManager.unscheduleAction(this.action.actionId);
405 			this.action.reset();
406 			this.begin();
407 		}
408 	}
409 };

411 /**
412  * @constructor
413  * @param {number} delay in seconds between frames
414  * @param {Array.<Object>} frames The frames of the animation
415  * @param {boolean=} loop Whether or not this animation should loop
416  * @param {chesterGL.Block=} block The block that will receive this action
417  * @extends {chesterGL.Action}
418  */
419 chesterGL.AnimateAction = function (delay, frames, loop, block) {
420 	this.delay = delay;
421 	var totalTime = delay * frames.length;
422 	if (loop === true) totalTime = -1;
423 	chesterGL.Action.call(this, totalTime, block);
424 	this.shouldLoop = (loop === true);
425 	this.frames = frames.slice(0);
426 };
427 goog.inherits(chesterGL.AnimateAction, chesterGL.Action);
428 
429 /**
430  * the current frame
431  * @type {number}
432  * @ignore
433  */
434 chesterGL.AnimateAction.prototype.currentFrame = 0;

436 /**
437  * The delay between frames
438  * @type {number}
439  * @ignore
440  */
441 chesterGL.AnimateAction.prototype.delay = 0.0;
442 
443 /**
444  * The total frames of the animation
445  * @type {Array.<goog.vec.Vec4.Type>}
446  * @ignore
447  */
448 chesterGL.AnimateAction.prototype.frames = null;
449 
450 /**
451  * Whether or not the animation should loop
452  * @type {boolean}
453  */
454 chesterGL.AnimateAction.prototype.shouldLoop = false;
455 
456 /**
457  * @param {number} delta
458  * @ignore
459  */
460 chesterGL.AnimateAction.prototype.update = function (delta) {
461 	chesterGL.Action.prototype.update.call(this, delta);
462 	var block = this.block;
463 	if (this.finished) {
464 		this.currentFrame = this.frames.length - 1;
465 		// set last frame
466 		block.setFrame(this.frames[this.currentFrame]);
467 	} else {
468 		if (this.elapsed >= this.delay * this.currentFrame) {
469 			block.setFrame(this.frames[this.currentFrame++]);
470 			if (this.currentFrame == this.frames.length) {
471 				if (this.shouldLoop) { this.currentFrame = 0; this.elapsed = 0; }
472 				else this.finished = true;
473 			}
474 		}
475 	}
476 };

478 /**
479  * @constructor
480  * Will rotate the block using a sin(t) function with the given
481  * Amplitude.
482  *
483  * @param {number} amplitude in radians of the wiggle
484  * @param {number} cycles the total number of cycles
485  * @param {number} totalTime the total duration (in milliseconds)
486  * @param {chesterGL.Block=} block The block that will receive this action
487  * @extends {chesterGL.Action}
488  */
489 chesterGL.WiggleAction = function (amplitude, cycles, totalTime, block) {
490 	this.amplitude = amplitude;
491 	this.cycles = cycles;
492 	chesterGL.Action.call(this, totalTime, block);
493 };
494 goog.inherits(chesterGL.WiggleAction, chesterGL.Action);
495 
496 /**
497  * @type {number} the amplitude of the wiggle
498  */
499 chesterGL.WiggleAction.prototype.amplitude = 0;
500 
501 /**
502  * @type {number} the number of cycles of the wiggle
503  */
504 chesterGL.WiggleAction.prototype.cycles = 0;
505 
506 /**
507  * @param {number} delta
508  * @ignore
509  */
510 chesterGL.WiggleAction.prototype.update = function (delta) {
511 	chesterGL.Action.prototype.update.call(this, delta);
512 	if (!this.finished) {
513 		// t' = t * cycles * 2 * PI / duration
514 		var _t = this.elapsed / 1000.0 * this.cycles * 2 * Math.PI / (this.totalTime / 1000.0);
515 		var sin = Math.sin(_t);
516 		// console.log("sin: " + sin + " - " + this.elapsed + " - " + t);
517 		this.block.setRotation(this.amplitude * sin);
518 	} else {
519 		// reset rotation
520 		this.block.setRotation(0);
521 	}
522 };

530 /**
531  * the list of scheduled actions
532  * @ignore
533  * @type {Array.<chesterGL.Action>}
534  * @private
535  */
536 chesterGL.ActionManager.scheduledActions_ = [];
537 
538 /**
539  * @type {Array.<chesterGL.Action>}
540  * @ignore
541  */
542 chesterGL.ActionManager.scheduledActionsToBeRemoved_ = [];
543 
544 /**
545  * @type {number}
546  * @ignore
547  */
548 chesterGL.ActionManager.internalIdCounter_ = 0;
549 
550 /**
551  * adds an action to the scheduler
552  *
553  * @param {chesterGL.Action} action
554  * @return {number} the actionId of the recently scheduled action
555  */
556 chesterGL.ActionManager.scheduleAction = function (action) {
557 	chesterGL.ActionManager.scheduledActions_.push(action);
558 	action.actionId = chesterGL.ActionManager.internalIdCounter_++;
559 	action.begin();
560 	return action.actionId;
561 };
562 
563 /**
564  * removes an action of the manager
565  * @param {number} actionId
566  */
567 chesterGL.ActionManager.unscheduleAction = function (actionId) {
568 	var len = chesterGL.ActionManager.scheduledActions_.length;
569 	for (var i=0; i < len; i++) {
570 		var a = chesterGL.ActionManager.scheduledActions_[i];
571 		if (a.actionId == actionId) {
572 			chesterGL.ActionManager.scheduledActionsToBeRemoved_.push(a);
573 			return;
574 		}
575 	}
576 };
577 
578 /**
579  * Iterate over all scheduled actions
580  * @param {number} delta number of miliseconds to run in all actions
581  * @ignore
582  */
583 chesterGL.ActionManager.tick = function (delta) {
584 	var i = 0,
585 		len = chesterGL.ActionManager.scheduledActions_.length,
586 		a = null;
587 	for (i=0; i < len; i++) {
588 		a = chesterGL.ActionManager.scheduledActions_[i];
589 		if (a.running) a.update(delta);
590 		if (a.finished) {
591 			chesterGL.ActionManager.scheduledActionsToBeRemoved_.push(a);
592 		}
593 	}
594 	// remove finished actions
595 	while((a = chesterGL.ActionManager.scheduledActionsToBeRemoved_.pop())) {
596 		var idx = chesterGL.ActionManager.scheduledActions_.indexOf(a);
597 		if (idx >= 0) {
598 			chesterGL.ActionManager.scheduledActions_.splice(idx, 1);
599 		}
600 	}
601 };

603 /**
604  * schedules an action to be run over this block
605  * @param {chesterGL.Action} action
606  */
607 chesterGL.Block.prototype.runAction = function (action) {
608 	action.block = this;
609 	chesterGL.ActionManager.scheduleAction(action);
