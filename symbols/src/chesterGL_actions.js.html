 31 /**
 32  * @constructor
 33  * @param {number} totalTime in seconds
 34  * @param {chesterGL.Block=} block The target block
 35  */
 36 chesterGL.Action = function (totalTime, block) {
 37 	this.totalTime = totalTime * 1000;
 38 	this.block = block;
 39 	this.elapsed = 0;
 40 };
 41 
 42 /**
 43  * The block to which this action will be applied
 44  * 
 45  * @type {chesterGL.Block|null|undefined}
 46  */
 47 chesterGL.Action.prototype.block = null;
 48 
 49 /**
 50  * The total time in seconds this action should take
 51  * (might not be relevant for all actions)
 52  * 
 53  * @type {number}
 54  */
 55 chesterGL.Action.prototype.totalTime = 0;
 56 
 57 /**
 58  * Current time in seconds of the action
 59  * (might not be relevant for all actions)
 60  * 
 61  * @type {number}
 62  */
 63 chesterGL.Action.prototype.elapsed = 0;
 64 
 65 /**
 66  * The current time of the action
 67  * @type {number}
 68  */
 69 chesterGL.Action.prototype.currentTime = 0;
 70 
 71 /**
 72  * The current time of the action
 73  * @type {boolean}
 74  */
 75 chesterGL.Action.prototype.finished = false;
 76 
 77 /**
 78  * is the action running?
 79  * @type {boolean}
 80  */
 81 chesterGL.Action.prototype.running = false;
 82 
 83 /**
 84  * This is the default update function (does nothing)
 85  * @param {number} delta
 86  */
 87 chesterGL.Action.prototype.update = function (delta) {
 88 	if (!this.running) {
 89 		this.begin();
 90 		this.running = true;
 91 	}
 92 	this.elapsed += delta;
 93 	if (this.totalTime > 0 && this.elapsed >= this.totalTime) {
 94 		this.finished = true;
 95 	}
 96 };
 97 
 98 /**
 99  * will be called the first time - usually overriden by subclasses
100  */
101 chesterGL.Action.prototype.begin = function () {
102 };
103 
104 /**
105  * @constructor
106  * @param {Array|Float32Array} finalPosition The final position (the initial position is the current one of the block)
107  * @param {number} totalTime The total time in seconds that this action should take
108  * @param {chesterGL.Block=} block The block that will execute this action
109  * @extends {chesterGL.Action}
110  */
111 chesterGL.MoveToAction = function (finalPosition, totalTime, block) {
112 	chesterGL.Action.call(this, totalTime, block);
113 	this.finalPosition = goog.vec.Vec3.createFloat32FromArray(finalPosition);
114 };
115 goog.inherits(chesterGL.MoveToAction, chesterGL.Action);
116 
117 /**
118  * @type {?goog.vec.Vec3.Type}
119  */
120 chesterGL.MoveToAction.prototype.finalPosition = null;
121 
122 /**
123  * @type {?goog.vec.Vec3.Type}
124  */
125 chesterGL.MoveToAction.prototype.startPosition = null;
126 
127 /**
128  * @type {goog.vec.Vec3.Type}
129  * @ignore
130  */
131 chesterGL.MoveToAction__tmp_pos = goog.vec.Vec3.createFloat32();
132 
133 /**
134  * @param {number} delta miliseconds from last time we updated
135  * @ignore
136  */
137 chesterGL.MoveToAction.prototype.update = function (delta) {
138 	chesterGL.Action.prototype.update.call(this, delta);
139 	var block = this.block;
140 	if (this.finished) {
141 		block.setPosition(this.finalPosition);
142 	} else {
143 		var t = Math.min(1, this.elapsed / this.totalTime);
144 		// console.log("t: " + t + "\t(" + dx + ")");
145 		goog.vec.Vec3.lerp(this.startPosition, this.finalPosition, t, chesterGL.MoveToAction__tmp_pos);
146 		block.setPosition(chesterGL.MoveToAction__tmp_pos);
147 	}
148 };
149 
150 /**
151  * just set the initial position
152  * @ignore
153  */
154 chesterGL.MoveToAction.prototype.begin = function () {
155 	if (!this.block) {
156 		throw "invalid move action! - now block";
157 	}
158 	this.startPosition = this.block.position;
159 };

161 /**
162  * @constructor
163  * @param {number} delay in seconds between frames
164  * @param {Array.<Object>} frames The frames of the animation
165  * @param {boolean=} loop Whether or not this animation should loop
166  * @param {chesterGL.Block=} block The block that will receive this action
167  * @extends {chesterGL.Action}
168  */
169 chesterGL.AnimateAction = function (delay, frames, loop, block) {
170 	this.delay = delay * 1000.0;
171 	var totalTime = this.delay * frames.length;
172 	if (loop === true) totalTime = -1;
173 	chesterGL.Action.call(this, totalTime, block);
174 	this.shouldLoop = (loop === true);
175 	this.frames = frames.slice(0);
176 };
177 goog.inherits(chesterGL.AnimateAction, chesterGL.Action);
178 
179 /**
180  * the current frame
181  * @type {number}
182  * @ignore
183  */
184 chesterGL.AnimateAction.prototype.currentFrame = 0;
185 
186 /**
187  * The delay between frames
188  * @type {number}
189  * @ignore
190  */
191 chesterGL.AnimateAction.prototype.delay = 0.0;
192 
193 /**
194  * The total frames of the animation
195  * @type {Array.<goog.vec.Vec4.Type>}
196  * @ignore
197  */
198 chesterGL.AnimateAction.prototype.frames = null;
199 
200 /**
201  * Whether or not the animation should loop
202  * @type {boolean}
203  */
204 chesterGL.AnimateAction.prototype.shouldLoop = false;
205 
206 /**
207  * @param {number} delta
208  * @ignore
209  */
210 chesterGL.AnimateAction.prototype.update = function (delta) {
211 	chesterGL.Action.prototype.update.call(this, delta);
212 	var block = this.block;
213 	if (this.finished) {
214 		this.currentFrame = this.frames.length - 1;
215 		this.finished = true;
216 		// set last frame
217 		block.setFrame(this.frames[this.currentFrame]);
218 	} else {
219 		if (this.elapsed >= this.delay * this.currentFrame) {
220 			block.setFrame(this.frames[this.currentFrame++]);
221 			if (this.currentFrame == this.frames.length) {
222 				if (this.shouldLoop) { this.currentFrame = 0; this.elapsed = 0; }
223 				else this.finished = true;
224 			}
225 		}
226 	}
227 };

229 /**
230  * global action manager
231  * @namespace
232  */
233 chesterGL.ActionManager = {};
234 
235 /**
236  * the list of scheduled actions
237  * @ignore
238  * @type {Array.<chesterGL.Action>}
239  * @private
240  */
241 chesterGL.ActionManager.scheduledActions_ = [];
242 
243 /**
244  * adds an action to the scheduler
245  * 
246  * @param {chesterGL.Action} action
247  */
248 chesterGL.ActionManager.scheduleAction = function (action) {
249 	chesterGL.ActionManager.scheduledActions_.push(action);
250 };
251 
252 /**
253  * Iterate over all scheduled actions
254  * @param {number} delta number of miliseconds to run in all actions
255  * @ignore
256  */
257 chesterGL.ActionManager.tick = function (delta) {
258 	var i = 0, len = chesterGL.ActionManager.scheduledActions_.length;
259 	for (i=0; i < len; i++) {
260 		var a = chesterGL.ActionManager.scheduledActions_[i];
261 		!a.finished && a.update(delta);
262 	}
263 };
264 
265 /**
266  * schedules an action to be run over this block
267  * @param {chesterGL.Action} action
268  */
269 chesterGL.Block.prototype.runAction = function (action) {
270 	action.block = this;
271 	chesterGL.ActionManager.scheduleAction(action);
272 };
