 26 goog.provide("chesterGL.LabelBlock");
 27 
 28 goog.require("chesterGL.core");
 29 goog.require("chesterGL.Block");
 30 
 31 /**
 32  * creates a new Label Block. Label blocks internally use a canvas to render the text.
 33  * These blocks should not be used for high-frecuency updated labels, for that you
 34  * should use bitmap fonts (not yet supported) or an off-screen DOM element.
 35  * Use this for static elements, like dialogues, menus, etc.
 36  *
 37  * @constructor
 38  * @param {string} text the string to render
 39  * @param {string=} font the name/family of the font, defaults to "20pt sans-serif"
 40  * @param {string=} fillStyle the style of the font, defaults to 'white'
 41  * @extends {chesterGL.Block}
 42  */
 43 chesterGL.LabelBlock = function (text, font, fillStyle) {
 44 	font = font || "20pt sans-serif";
 45 	fillStyle = fillStyle || "White";
 46 	var canvas = document.createElement("canvas");
 47 	this.canvas = canvas;
 48 	var context = canvas.getContext('2d');
 49 	this.context = context;
 50 	this.font = font;
 51 	this.fillStyle = fillStyle;
 52 
 53 	this.texture = Date.now() + ".canvas";
 54 	if (!chesterGL.assets['texture']) chesterGL.assets['texture'] = {};
 55 	chesterGL.assets['texture'][this.texture] = canvas;
 56 
 57 	chesterGL.Block.call(this, this.resetCanvas(text));
 58 	this.setText(text, true);
 59 	this.program = chesterGL.Block.PROGRAM['TEXTURE'];
 60 };
 61 goog.inherits(chesterGL.LabelBlock, chesterGL.Block);
 62 
 63 /**
 64  * The canvas where we will draw
 65  * @type {?Element}
 66  * @ignore
 67  */
 68 chesterGL.LabelBlock.prototype.canvas = null;
 69 
 70 /**
 71  * Our 2d context for drawing the text
 72  * @type {?CanvasRenderingContext2D}
 73  * @ignore
 74  */
 75 chesterGL.LabelBlock.prototype.context = null;
 76 
 77 /**
 78  * Whether or not we need to redraw the text
 79  * @type {boolean}
 80  */
 81 chesterGL.LabelBlock.prototype.needsTextUpdate = false;
 82 
 83 /**
 84  * the actual string to draw
 85  * @ignore
 86  * @type {string}
 87  */
 88 chesterGL.LabelBlock.prototype.text = "";
 89 
 90 /**
 91  * the height of the text (aprox.)
 92  * @ignore
 93  * @type {number}
 94  */
 95 chesterGL.LabelBlock.prototype.textHeight = 0;
 96 
 97 /**
 98  * the font that will be used to render the text
 99  * @type {string}
100  */
101 chesterGL.LabelBlock.prototype.font = "";
102 
103 /**
104  * the style used to render the text. Need to set the text again
105  * to actually change the style
106  * @type {string}
107  */
108 chesterGL.LabelBlock.prototype.fillStyle = "";
109 
110 /**
111  * sets the text to be displayed. This will reset the underlying canvas and resize it to the size of the text
112  * @param {string} text
113  * @param {boolean=} onlyDraw if true, then just draw the text on the canvas
114  */
115 chesterGL.LabelBlock.prototype.setText = function (text, onlyDraw) {
116 	this.text = text;
117 	if (onlyDraw) {
118 		this.drawText();
119 	} else {
120 		this.setFrame(this.resetCanvas());
121 		this.needsTextUpdate = true;
122 	}
123 };
124 
125 /**
126  * redraws the text on the offscreen canvas (clearing it first)
127  * @ignore
128  */
129 chesterGL.LabelBlock.prototype.drawText = function () {
130 	var cx = this.context;
131 	var canvas = this.canvas;
132 	cx.clearRect(0, 0, canvas.width, canvas.height);
133 	cx.fillText(this.text, 0, canvas.height * 0.8);
134 	chesterGL.defaultTextureHandler(/** @type {string} */(this.texture), canvas);
135 	this.needsTextUpdate = false;
136 };
137 
138 /**
139  * resets the canvas: sets the style, measures the text, changes the canvas size.
140  * Returns an array that can be passed to setFrame().
141  * @ignore
142  * @param {string=} text optional text to set
143  * @returns {Array}
144  */
145 chesterGL.LabelBlock.prototype.resetCanvas = function (text) {
146 	var cx = this.context;
147 	var canvas = this.canvas;
148 	cx.font = this.font;
149 	cx.fillStyle = this.fillStyle;
150 	if (text) {
151 		this.text = text;
152 	}
153 
154 	if (this.textHeight === 0) {
155 		cx.save();
156 		cx.rotate(90);
157 		this.textHeight = cx.measureText("M").width * 1.25;
158 		cx.restore();
159 	}
160 	var width = cx.measureText(this.text).width;
161 
162 	canvas.width = width;
163 	canvas.height = this.textHeight;
164 	cx.font = this.font;
165 	cx.fillStyle = this.fillStyle;
166 	return [0, 0, width, this.textHeight];
167 };
168 
169 /**
170  * override visit, update offscreen canvas if needed
171  * @ignore
172  */
173 chesterGL.LabelBlock.prototype.visit = function () {
174 	if (this.needsTextUpdate) {
175 		this.drawText();
176 	}
177 	chesterGL.Block.prototype.visit.call(this);
178 };
179 
180 goog.exportSymbol('chesterGL.LabelBlock', chesterGL.LabelBlock);
181 goog.exportProperty(chesterGL.LabelBlock.prototype, 'setText', chesterGL.LabelBlock.prototype.setText);
