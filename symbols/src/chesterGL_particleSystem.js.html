 35 var BUFFER_ELEMENTS = 10;

 37 /**
 38  * BUFFER_ELEMENTS * 4 (bytes) ~> BUFFER_ELEMENTS floats == 1 (lifetime) + 1 (start time) + 1 (start size) + 1 (end size) + 3 (start pos) + 3 (end pos)
 39  * @const {number}
 40  * @ignore
 41  */
 42 var PARTICLE_SIZE = 40;
 43 
 44 /**
 45  * @param {goog.vec.Vec3.Type} original The original vector
 46  * @param {goog.vec.Vec3.Type} variance the variance for every coordinate in the original vector
 47  * @return {goog.vec.Vec3.Type}
 48  * @ignore
 49  */
 50 function randomVec3(original, variance) {
 51 	var vec = goog.vec.Vec3.createFloat32();
 52 	if (variance) {
 53 		vec[0] = original[0] + (variance[0] * 2 * Math.random());
 54 		vec[1] = original[1] + (variance[1] * 2 * Math.random());
 55 		vec[2] = original[2] + (variance[2] * 2 * Math.random());
 56 	} else {
 57 		vec[0] = original[0];
 58 		vec[1] = original[1];
 59 		vec[2] = original[2];
 60 	}
 61 	return vec;
 62 }

 64 /**
 65  * @constructor
 66  * @param {Object} properties
 67  * @extends chesterGL.Block
 68  */
 69 chesterGL.ParticleSystem = function (properties) {
 70 	chesterGL.Block.call(this, null, chesterGL.Block.TYPE.PARTICLE);
 71 	var _this = this;
 72 	chesterGL.loadAsset('texture', properties['texture'], function () {
 73 		_this.loadProperties(properties);
 74 	});
 75 };
 76 goog.inherits(chesterGL.ParticleSystem, chesterGL.Block);

 84 /**
 85  * Load the shaders for the particle system
 86  */
 87 chesterGL.ParticleSystem.loadShaders = function () {
 88 	chesterGL.initShader("particles", function (program) {
 89 		var gl = chesterGL.gl;
 90 		program.mvpMatrixUniform = gl.getUniformLocation(program, "uMVPMatrix");
 91 		program.uSampler         = gl.getUniformLocation(program, "uSampler");
 92 		program.u_time           = gl.getUniformLocation(program, "u_time");
 93 		program.u_startColor     = gl.getUniformLocation(program, "u_startColor");
 94 		program.u_endColor       = gl.getUniformLocation(program, "u_endColor");
 95 		program.attribs = {
 96 			'a_startPosition': gl.getAttribLocation(program, 'a_startPosition'),
 97 			'a_lifetime'  : gl.getAttribLocation(program, 'a_lifetime'),
 98 			'a_startTime'  : gl.getAttribLocation(program, 'a_startTime'),
 99 			'a_startSize'  : gl.getAttribLocation(program, 'a_startSize'),
100 			'a_endSize'  : gl.getAttribLocation(program, 'a_endSize'),
101 			'a_speed'        : gl.getAttribLocation(program, 'a_speed')
102 		};
103 					
104 		// test for errors on gl
105 		var error = gl.getError();
106 		if (error != 0) {
107 			console.log("gl error: " + error);
108 		}
109 	});
110 	chesterGL.ParticleSystem._shadersLoaded = true;
111 }

238 /**
239  * @param {Object} properties
240  */
241 chesterGL.ParticleSystem.prototype.loadProperties = function (properties) {
242 	this.program = -1;
243 	if (!chesterGL.ParticleSystem._shadersLoaded) {
244 		chesterGL.ParticleSystem.loadShaders();
245 	}
246 	this.particleTexture = properties['texture'];
247 	this.maxParticles = properties['maxParticles'];
248 	this.duration = parseFloat(properties['duration']) * 1000.0;
249 	this.lifetime = parseFloat(properties['lifetime']) * 1000.0;
250 	this.lifetimeVariance = parseFloat(properties['lifetimeVariance']) * 1000.0;
251 	this.startColor = goog.vec.Vec4.createFloat32FromArray(properties['startColor']);
252 	this.positionVariance = goog.vec.Vec3.createFloat32FromArray(properties['positionVariance']);
253 	this.endColor = goog.vec.Vec4.createFloat32FromArray(properties['endColor']);
254 	this.particleSpeed = goog.vec.Vec3.createFloat32FromArray(properties['speed']);
255 	this.particleSpeedVariance = goog.vec.Vec3.createFloat32FromArray(properties['speedVariance']);
256 	this.startSize = parseFloat(properties['startSize']);
257 	this.startSizeVariance = parseFloat(properties['startSizeVariance']);
258 	this.endSize = parseFloat(properties['endSize']);
259 	this.endSizeVariance = parseFloat(properties['endSizeVariance']);
260 	this.elapsedTime = 0;
261 	this.blendOptions = properties['blendOptions'].slice(0); // copy the array
262 	this.running = true;
263 	
264 	this.glBuffer = chesterGL.gl.createBuffer();
265 	this.glBufferData = new Float32Array(this.maxParticles * BUFFER_ELEMENTS);
266 	this.resetParticles();
267 }

269 /**
270  * adds a new particle (sets the lifetime in the data sent to the shader)
271  */
272 chesterGL.ParticleSystem.prototype.addParticle = function () {
273 	var lifespan = Math.abs(this.lifetime + this.lifetimeVariance * (Math.random() * 2 - 1));
274 	this.initParticle(this.particleCount, lifespan, this.elapsedTime);
275 	this.particleCount++;
276 	this.particleAdded = true;
277 }
278 
279 /**
280  * @param {number} idx
281  * @param {number=} lifetime
282  * @param {number=} startTime
283  */
284 chesterGL.ParticleSystem.prototype.initParticle = function (idx, lifetime, startTime) {
285 	var d = this.glBufferData;
286 	lifetime = lifetime || -1.0;
287 	startTime = startTime || 0.0;
288 	
289 	// lifetime, start time, start size, end size
290 	d[idx * BUFFER_ELEMENTS + 0] = lifetime;
291 	d[idx * BUFFER_ELEMENTS + 1] = startTime;
292 	d[idx * BUFFER_ELEMENTS + 2] = this.startSize + this.startSizeVariance * (Math.random() * 2 - 1);
293 	d[idx * BUFFER_ELEMENTS + 3] = this.endSize + this.endSizeVariance * (Math.random() * 2 - 1);
294 	
295 	// speed
296 	d[idx * BUFFER_ELEMENTS + 4] = this.particleSpeed[0] + this.particleSpeedVariance[0] * (Math.random() * 2 - 1);
297 	d[idx * BUFFER_ELEMENTS + 5] = this.particleSpeed[1] + this.particleSpeedVariance[1] * (Math.random() * 2 - 1);
298 	d[idx * BUFFER_ELEMENTS + 6] = this.particleSpeed[2] + this.particleSpeedVariance[2] * (Math.random() * 2 - 1);
299 	
300 	// start position
301 	d[idx * BUFFER_ELEMENTS + 7] = (Math.random() * 2 - 1) * this.positionVariance[0];
302 	d[idx * BUFFER_ELEMENTS + 8] = (Math.random() * 2 - 1) * this.positionVariance[1];
303 	d[idx * BUFFER_ELEMENTS + 9] = (Math.random() * 2 - 1) * this.positionVariance[2];
304 }

306 /**
307  * reset particle data - this is slow!
308  */
309 chesterGL.ParticleSystem.prototype.resetParticles = function () {
310 	var program = chesterGL.selectProgram("particles");
311 	var gl = chesterGL.gl;
312 	for (var i = 0; i < this.maxParticles; i++) {
313 		this.initParticle(i);
314 	}
315 	gl.uniform4fv(program.u_startColor, /** @type {Float32Array} */(this.startColor));
316 	gl.uniform4fv(program.u_endColor  , /** @type {Float32Array} */(this.endColor));
317 	gl.uniform1i(program.uSampler, 0);
318 
319 	this.sendParticleData(program);
320 	
321 	this.particleCount = this.emissionCounter = 0;
322 	// how many particles are emitted per second
323 	this.emissionRate = this.maxParticles / Math.abs(this.lifetime);
324 }
325 
326 /**
327  * will send the particle data to the gpu
328  * @param {WebGLProgram} program
329  */
330 chesterGL.ParticleSystem.prototype.sendParticleData = function (program) {
331 	var gl = chesterGL.gl;
332 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
333 	gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
334 }

336 /**
337  * @type {Float32Array}
338  * @ignore
339  */
340 var _ps_tmp = new Float32Array(BUFFER_ELEMENTS);
341 
342 /**
343  * The update loop for the particle system
344  * @param {number} delta
345  */
346 chesterGL.ParticleSystem.prototype.update = function (delta) {
347 	var program = chesterGL.selectProgram("particles");
348 	if (!program) {
349 		return;
350 	}
351 	this.elapsedTime += delta;
352 
353 	// how many seconds until the next particle
354 	var rate = 1.0 / this.emissionRate;
355 	this.emissionCounter += delta;
356 	while (this.particleCount < this.maxParticles && this.emissionCounter > rate && this.running) {
357 		this.addParticle();
358 		this.emissionCounter -= rate;
359 	}
360 
361 	for (var i = 0; i < this.maxParticles; i++) {
362 		var buffer = this.glBufferData;
363 		var idx = i * BUFFER_ELEMENTS;
364 		// if expired, move the (buffer) particle ahead
365 		if (buffer[idx] > 0 && (buffer[idx] + buffer[idx+1]) <= this.elapsedTime && i != this.particleCount - 1) {
366 			// copy the particle into the tmp buffer and invalidate
367 			var tmp = buffer.subarray(idx, idx + BUFFER_ELEMENTS);
368 			_ps_tmp.set(tmp);
369 			_ps_tmp[0] = -1.0;
370 			// shift the array from idx to particleCount
371 			tmp = buffer.subarray(idx + BUFFER_ELEMENTS, this.particleCount * BUFFER_ELEMENTS);
372 			buffer.set(tmp, idx);
373 			// copy the old particle in the last spot
374 			buffer.set(_ps_tmp, (this.particleCount-1) * BUFFER_ELEMENTS);
375 			// decrease the particle count
376 			this.particleCount --;
377 		}
378 	}
379 
380 	if (this.duration > 0 && this.elapsedTime > this.duration) {
381 		this.running = false;
382 	}
383 }

385 chesterGL.ParticleSystem.prototype.render = function () {
386 	var program = chesterGL.selectProgram("particles");
387 	if (!program) {
388 		return;
389 	}
390 	var gl = chesterGL.gl;
391 	var texture = chesterGL.getAsset('texture', this.particleTexture);
392 	
393 	gl.enable(gl.BLEND);
394 	gl.blendFunc(gl[this.blendOptions[0]], gl[this.blendOptions[1]]);
395 	
396 	if (this.particleAdded) {
397 		this.sendParticleData(program);
398 		this.particleAdded = false;
399 	}
400 	
401 	// send the elapsed time
402 	gl.uniform1f(program.u_time, this.elapsedTime);
403 	
404 	// activate the texture
405 	gl.activeTexture(gl.TEXTURE0);
406 	gl.bindTexture(gl.TEXTURE_2D, texture.tex);
407 	
408 	// bind buffer
409 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
410 	gl.vertexAttribPointer(program.attribs['a_lifetime']     , 3, gl.FLOAT, false, PARTICLE_SIZE, 0);
411 	gl.vertexAttribPointer(program.attribs['a_startTime']    , 3, gl.FLOAT, false, PARTICLE_SIZE, 4);
412 	gl.vertexAttribPointer(program.attribs['a_startSize']    , 3, gl.FLOAT, false, PARTICLE_SIZE, 8);
413 	gl.vertexAttribPointer(program.attribs['a_endSize']      , 3, gl.FLOAT, false, PARTICLE_SIZE, 12);
414 	gl.vertexAttribPointer(program.attribs['a_speed']        , 3, gl.FLOAT, false, PARTICLE_SIZE, 16);
415 	gl.vertexAttribPointer(program.attribs['a_startPosition'], 3, gl.FLOAT, false, PARTICLE_SIZE, 28);
416 	
417 	// and draw:
418 	var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
419 	if (transformDirty) {
420 		goog.vec.Mat4.multMat(chesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
421 	}
422 	gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
423 	gl.drawArrays(gl.POINTS, 0, this.maxParticles);
424 }
