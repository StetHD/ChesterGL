 32 /**
 33  * creates a new block. Pass the rect if you want to set the frame at
 34  * creation time.
 35  * 
 36  * @constructor
 37  * @param {Object.<string,number>|string=} rect
 38  * @param {number=} type
 39  * @param {chesterGL.Block=} parent
 40  */
 41 chesterGL.Block = function (rect, type, parent) {
 42 	this.type = type || chesterGL.Block.TYPE['STANDALONE'];
 43 	if (parent) {
 44 		this.parent = parent;
 45 	}
 46 
 47 	this.children = [];
 48 	this.program = chesterGL.Block.PROGRAM['DEFAULT'];
 49 
 50 	if (rect) {
 51 		if (typeof(rect) === 'string') {
 52 			/** @type {chesterGL.Block.frameType} */ var f = chesterGL.BlockFrames.getFrame(rect);
 53 			this.setTexture(f.texture);
 54 			this.setFrame(f.frame);
 55 		} else {
 56 			this.setFrame(rect);
 57 		}
 58 	}
 59 	// set default color
 60 	if (this.type == chesterGL.Block.TYPE['STANDALONE']) {
 61 		this.setColor([1, 1, 1, 1]);
 62 	}
 63 
 64 	if (chesterGL.webglMode && this.type == chesterGL.Block.TYPE['STANDALONE'] && (!parent || parent.type != chesterGL.Block.TYPE['BLOCKGROUP'])) {
 65 		var gl = chesterGL.gl;
 66 		// just a single buffer for all data (a 4x"quad")
 67 		this.glBuffer = gl.createBuffer();
 68 		this.glBufferData = new Float32Array(chesterGL.Block.BUFFER_SIZE);
 69 	}
 70 
 71 	// always create the mvMatrix
 72 	this.mvMatrix = mat4.create();
 73 	this.mvpMatrix = mat4.create();
 74 	mat4.identity(this.mvMatrix);
 75 
 76 	// create the remove/add lists
 77 	this._scheduledAdd = [];
 78 	this._scheduledRemove = [];
 79 };

294 /**
295  * sets the frame for this block
296  * 
297  * @param {quat4} newFrame
298  */
299 chesterGL.Block.prototype.setFrame = function (newFrame) {
300 	if (typeof newFrame === "string") {
301 		// just get the cached frame
302 		newFrame = chesterGL.BlockFrames.getFrame(/** @type {string} */(newFrame)).frame;
303 	}
304 	this.frame = quat4.create(newFrame);
305 	this.setContentSize(newFrame[2], newFrame[3]);
306 	this.isFrameDirty = true;
307 }

349 /**
350  * sets the texture of the block - the texture will be loaded if needed
351  * @param {string} texturePath
352  */
353 chesterGL.Block.prototype.setTexture = function (texturePath) {
354 	this.texture = texturePath;
355 	// force program to texture program
356 	this.program = chesterGL.Block.PROGRAM['TEXTURE'];
357 	var block = this;
358 	chesterGL.loadAsset("texture", texturePath, function (t) {
359 		// set the default frame for all our blocks (if it's not set)
360 		if (!block.contentSize) {
361 			block.setContentSize(t.width, t.height);
362 		}
363 		if (!block.frame) {
364 			block.setFrame([0, 0, t.width, t.height]);
365 		}
366 	});
367 };

386 /**
387  * Adds a block as a child. If you add the block while in a visit of the parent block,
388  * the child will be scheduled to be added after the visit.
389  * 
390  * @param {chesterGL.Block} block
391  */
392 chesterGL.Block.prototype.addChild = function (block) {
393 	if (block.parent) {
394 		throw "can't add a block twice!";
395 	}
396 	if (this._inVisit) {
397 		this._scheduledAdd.push(block);
398 	} else {
399 		this.children.push(block);
400 		block.parent = this;
401 	}
402 };
403 
404 /**
405  * removes a block from the children list. If you remove the child while in a visit of the parent block,
406  * the child will be scheduled to be removed after the visit.
407  * 
408  * @param {chesterGL.Block} block
409  */
410 chesterGL.Block.prototype.removeChild = function (block) {
411 	if (!block.parent || block.parent != this) {
412 		throw "not our child!";
413 	}
414 	if (this._inVisit) {
415 		this._scheduledRemove.push(block);
416 	} else {
417 		var idx = this.children.indexOf(block);
418 		if (idx >= 0) {
419 			this.children.splice(idx,1);
420 		}
421 	}
422 }

424 /**
425  * actually performs the transformation
426  */
427 chesterGL.Block.prototype.transform = function () {
428 	var gl = chesterGL.gl;
429 	var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
430 	if (transformDirty) {
431 		mat4.identity(this.mvMatrix);
432 		mat4.translate(this.mvMatrix, this.position);
433 		mat4.rotate(this.mvMatrix, this.rotation * (chesterGL.webglMode ? -1 : 1), [0, 0, 1]);
434 		mat4.scale(this.mvMatrix, [this.scale, this.scale, 1]);
435 		// concat with parent's transform
436 		var ptransform = (this.parent ? this.parent.mvMatrix : null);
437 		if (ptransform) {
438 			mat4.multiply(ptransform, this.mvMatrix, this.mvMatrix);
439 		}
440 	}
441 
442 	// bail out if we're a block group
443 	if (this.type == chesterGL.Block.TYPE['BLOCKGROUP']) {
444 		return;
445 	}
446 
447 	var bufferData = this.glBufferData;
448 	var inBlockGroup = this.parent && this.parent.type == chesterGL.Block.TYPE['BLOCKGROUP'];
449 
450 	if (chesterGL.webglMode) {
451 		if (!inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
452 			gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
453 		}
454 		if (this.isFrameDirty || (inBlockGroup && this.isTransformDirty)) {
455 			// NOTE
456 			// the tex coords and the frame coords need to match. Otherwise you get a distorted image
457 			var offset = 9;
458 			var hw = this.contentSize.width * 0.5, hh = this.contentSize.height * 0.5;
459 			var _idx = this.baseBufferIndex * chesterGL.Block.BUFFER_SIZE;
460 			var z = this.position[2];
461 
462 			// NOTE
463 			// this is going to be slow :P
464 			// this is using the matrix to transform the vertex data
465 			if (inBlockGroup) {
466 				var tr = [ hw,  hh, 0],
467 					tl = [-hw,  hh, 0],
468 					br = [ hw, -hh, 0],
469 					bl = [-hw, -hh, 0];
470 				mat4.multiplyVec3(this.mvMatrix, tr);
471 				mat4.multiplyVec3(this.mvMatrix, tl);
472 				mat4.multiplyVec3(this.mvMatrix, bl);
473 				mat4.multiplyVec3(this.mvMatrix, br);
474 
475 				bufferData[_idx           ] =  bl[0]; bufferData[_idx + 1           ] =  bl[1]; bufferData[_idx + 2           ] = z;
476 				bufferData[_idx +   offset] =  tl[0]; bufferData[_idx + 1 +   offset] =  tl[1]; bufferData[_idx + 2 +   offset] = z;
477 				bufferData[_idx + 2*offset] =  br[0]; bufferData[_idx + 1 + 2*offset] =  br[1]; bufferData[_idx + 2 + 2*offset] = z;
478 				bufferData[_idx + 3*offset] =  tr[0]; bufferData[_idx + 1 + 3*offset] =  tr[1]; bufferData[_idx + 2 + 3*offset] = z;
479 			} else {
480 				bufferData[_idx           ] = -hw; bufferData[_idx + 1           ] = -hh; bufferData[_idx + 2           ] = 0;
481 				bufferData[_idx +   offset] = -hw; bufferData[_idx + 1 +   offset] =  hh; bufferData[_idx + 2 +   offset] = 0;
482 				bufferData[_idx + 2*offset] =  hw; bufferData[_idx + 1 + 2*offset] = -hh; bufferData[_idx + 2 + 2*offset] = 0;
483 				bufferData[_idx + 3*offset] =  hw; bufferData[_idx + 1 + 3*offset] =  hh; bufferData[_idx + 2 + 3*offset] = 0;
484 			}
485 
486 			if (this.program == chesterGL.Block.PROGRAM['TEXTURE']) {
487 				var tex = chesterGL.getAsset("texture", this.texture);
488 				var texW = tex.width,
489 					texH = tex.height;
490 				var l = this.frame[0] / texW,
491 					b = this.frame[1] / texH,
492 					w = this.frame[2] / texW,
493 					h = this.frame[3] / texH;
494 				_idx += 3;
495 				bufferData[_idx           ] = l  ; bufferData[_idx+1           ] = b;
496 				bufferData[_idx +   offset] = l  ; bufferData[_idx+1 +   offset] = b+h;
497 				bufferData[_idx + 2*offset] = l+w; bufferData[_idx+1 + 2*offset] = b;
498 				bufferData[_idx + 3*offset] = l+w; bufferData[_idx+1 + 3*offset] = b+h;
499 			}
500 		}
501 		if (this.isColorDirty) {
502 			_idx = 5 + this.baseBufferIndex * chesterGL.Block.BUFFER_SIZE;
503 			var color = this.color;
504 			var opacity = this.opacity;
505 			for (var i=0; i < 4; i++) {
506 				bufferData[_idx     + offset*i] = color[0] * opacity;
507 				bufferData[_idx + 1 + offset*i] = color[1] * opacity;
508 				bufferData[_idx + 2 + offset*i] = color[2] * opacity;
509 				bufferData[_idx + 3 + offset*i] = color[3] * opacity;
510 			}
511 		}
512 		if (chesterGL.webglMode && !inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
513 			gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
514 		}
515 	}
516 }

518 /**
519  * prepares the block for the rendering (transforms if necessary)
520  */
521 chesterGL.Block.prototype.visit = function () {
522 	this._inVisit = true;
523 	if (this.update) {
524 		this.update(chesterGL.delta);
525 	}
526 	if (!this.visible) {
527 		this._inVisit = false;
528 		return;
529 	}
530 	this.transform();
531 
532 	var children = this.children;
533 	var len = children.length;
534 	for (var i=0; i < len; i++) {
535 		children[i].visit();
536 	}
537 
538 	// render this block if not in a block group
539 	if (!this.parent || this.parent.type != chesterGL.Block.TYPE['BLOCKGROUP']) {
540 		this.render();
541 	}
542 	// reset our dirty markers
543 	this.isFrameDirty = this.isColorDirty = this.isTransformDirty = false;
544 	this._inVisit = false;
545 
546 	// do we have blocks scheduled to be removed/added?
547 	var b;
548 	while (b = this._scheduledAdd.shift()) {
549 		this.addChild(b);
550 	}
551 	while (b = this._scheduledRemove.shift()) {
552 		this.removeChild(b);
553 	}
554 }

556 /**
557  * render (only will work for non-blockgroup blocks)
558  */
559 chesterGL.Block.prototype.render = function () {
560 	if (this.type == chesterGL.Block.TYPE['BLOCKGROUP']) {
561 		throw "Cannot call render on a BlockGroup block!";
562 	}
563 	// dummy render for scene blocks
564 	if (this.type == chesterGL.Block.TYPE['SCENE']) {
565 		return;
566 	}
567 
568 	if (chesterGL.webglMode) {
569 		var gl = chesterGL.gl;
570 		// select current shader
571 		var program = chesterGL.selectProgram(chesterGL.Block.PROGRAM_NAME[this.program]);
572 
573 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
574 		var texOff = 3 * 4,
575 			colorOff = texOff + 2 * 4,
576 			stride = chesterGL.Block.QUAD_SIZE;
577 
578 		gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
579 		gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, colorOff);
580 
581 		if (this.program == chesterGL.Block.PROGRAM['DEFAULT']) {
582 			// no extra attributes for the shader
583 		} else if (this.program == chesterGL.Block.PROGRAM['TEXTURE']) {
584 			var texture = chesterGL.getAsset('texture', this.texture);
585 
586 			// pass the texture attributes
587 			gl.vertexAttribPointer(program.attribs['textureCoordAttribute'], 2, gl.FLOAT, false, stride, texOff);
588 
589 			gl.activeTexture(gl.TEXTURE0);
590 			gl.bindTexture(gl.TEXTURE_2D, texture.tex);
591 			gl.uniform1i(program.samplerUniform, 0);
592 			gl.activatedTexture = true;
593 		}
594 
595 		// set the matrix uniform (the multiplied model view projection matrix)
596 		var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
597 		if (transformDirty) {
598 			mat4.multiply(chesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
599 		}
600 		gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
601 		gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
602 	} else {
603 		var gl = chesterGL.offContext;
604 		// canvas drawing api - we only draw textures
605 		if (this.program == chesterGL.Block.PROGRAM.TEXTURE) {
606 			var m = this.mvMatrix;
607 			var texture = chesterGL.getAsset('texture', this.texture);
608 			gl.globalAlpha = this.opacity;
609 			gl.setTransform(m[0], m[1], m[4], m[5], m[12], gl.viewportHeight - m[13]);
610 			var w = this.contentSize.width, h = this.contentSize.height;
611 			var frame = this.frame;
612 			gl.drawImage(texture, frame[0], texture.height - (frame[1] + h), frame[2], frame[3], -w/2, -h/2, w, h);
613 		}
614 	}
615 }
