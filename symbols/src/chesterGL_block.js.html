 32 /**
 33  * creates a new block. Pass the rect if you want to set the frame at
 34  * creation time.
 35  * 
 36  * @constructor
 37  * @param {goog.vec.Vec4.Vec4Like|string=} rect
 38  * @param {number=} type
 39  * @param {chesterGL.Block=} parent
 40  */
 41 chesterGL.Block = function (rect, type, parent) {
 42 	this.type = type || chesterGL.Block.TYPE['STANDALONE'];
 43 	if (parent) {
 44 		this.parent = parent;
 45 	}
 46 
 47 	this.children = [];
 48 	this.program = chesterGL.Block.PROGRAM['DEFAULT'];
 49 
 50 	if (rect) {
 51 		this.setFrame(rect);
 52 	}
 53 	// set default color
 54 	if (this.type == chesterGL.Block.TYPE['STANDALONE']) {
 55 		this.setColor([1, 1, 1, 1]);
 56 	}
 57 
 58 	if (chesterGL.webglMode && this.type == chesterGL.Block.TYPE['STANDALONE'] && (!parent || parent.type != chesterGL.Block.TYPE['BLOCKGROUP'])) {
 59 		var gl = chesterGL.gl;
 60 		// just a single buffer for all data (a 4x"quad")
 61 		this.glBuffer = gl.createBuffer();
 62 		this.glBufferData = new Float32Array(chesterGL.Block.BUFFER_SIZE);
 63 	}
 64 
 65 	// always create the mvMatrix
 66 	this.mvMatrix = goog.vec.Mat4.createFloat32();
 67 	this.mvpMatrix = goog.vec.Mat4.createFloat32();
 68 	this.mvMatrix = /** @type {Float32Array} */(goog.vec.Mat4.makeIdentity(goog.vec.Mat4.createFloat32()));
 69 
 70 	// create the remove/add lists
 71 	this._scheduledAdd = [];
 72 	this._scheduledRemove = [];
 73 };

322 		var dbg = new chesterGL.PrimitiveBlock(1, 1);
323 		this.addChild(dbg);

327 			var poly = [[-w, -h, 0], [-w, h, 0], [w, h, 0], [w, -h, 0]];

334 /**
335  * sets the frame for this block
336  * 
337  * @param {Array|Float32Array|string} newFrame
338  */
339 chesterGL.Block.prototype.setFrame = function (newFrame) {
340 	if (typeof newFrame === "string") {
341 		// just get the cached frame
342 		var tmpFrame = chesterGL.BlockFrames.getFrame(newFrame);
343 		newFrame = tmpFrame.frame;
344 		this.setTexture(tmpFrame.texture);
345 	}
346 	this.frame = goog.vec.Vec4.createFloat32FromArray(newFrame);
347 	this.setContentSize(newFrame[2], newFrame[3]);
348 	this.isFrameDirty = true;
349 };

395 /**
396  * sets the texture of the block - the texture will be loaded if needed
397  * @param {string} texturePath
398  */
399 chesterGL.Block.prototype.setTexture = function (texturePath) {
400 	this.texture = texturePath;
401 	// force program to texture program
402 	this.program = chesterGL.Block.PROGRAM['TEXTURE'];
403 	if (chesterGL.debugSprite) { this.addDebugNode(); }
404 	var block = this;
405 	chesterGL.loadAsset("texture", texturePath, function (t) {
406 		// set the default frame for all our blocks (if it's not set)
407 		if (!block.contentSize) {
408 			block.setContentSize(t.width, t.height);
409 		}
410 		if (!block.frame) {
411 			block.setFrame([0, 0, t.width, t.height]);
412 		}
413 	});
414 };

433 /**
434  * Adds a block as a child. If you add the block while in a visit of the parent block,
435  * the child will be scheduled to be added after the visit.
436  * 
437  * @param {chesterGL.Block} block
438  */
439 chesterGL.Block.prototype.addChild = function (block) {
440 	if (block.parent) {
441 		throw "can't add a block twice!";
442 	}
443 	if (this._inVisit) {
444 		this._scheduledAdd.push(block);
445 	} else {
446 		this.children.push(block);
447 		block.parent = this;
448 	}
449 };
450 
451 /**
452  * removes a block from the children list. If you remove the child while in a visit of the parent block,
453  * the child will be scheduled to be removed after the visit.
454  * 
455  * @param {chesterGL.Block} block
456  */
457 chesterGL.Block.prototype.removeChild = function (block) {
458 	if (!block.parent || block.parent != this) {
459 		throw "not our child!";
460 	}
461 	if (this._inVisit) {
462 		this._scheduledRemove.push(block);
463 	} else {
464 		var idx = this.children.indexOf(block);
465 		if (idx >= 0) {
466 			this.children.splice(idx,1);
467 		}
468 	}
469 }

471 /**
472  * actually performs the transformation
473  * @ignore
474  */
475 chesterGL.Block.prototype.transform = function () {
476 	var gl = chesterGL.gl;
477 	var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
478 	if (transformDirty) {
479 		this.mvMatrix = /** @type {Float32Array} */(goog.vec.Mat4.makeIdentity(this.mvMatrix));
480 		goog.vec.Mat4.translate(this.mvMatrix, this.position[0], this.position[1], this.position[2]);
481 		goog.vec.Mat4.rotate(this.mvMatrix, this.rotation * (chesterGL.webglMode ? -1 : 1), 0, 0, 1);
482 		goog.vec.Mat4.scale(this.mvMatrix, this.scale, this.scale, 1);
483 		// concat with parent's transform
484 		var ptransform = (this.parent ? this.parent.mvMatrix : null);
485 		if (ptransform) {
486 			goog.vec.Mat4.multMat(ptransform, this.mvMatrix, this.mvMatrix);
487 		}
488 	}
489 
490 	// bail out if we're a block group or a primitive block
491 	if (this.type == chesterGL.Block.TYPE['BLOCKGROUP'] || this.type == chesterGL.Block.TYPE['PRIMITIVE']) {
492 		return;
493 	}
494 
495 	var bufferData = this.glBufferData;
496 	var inBlockGroup = this.parent && this.parent.type == chesterGL.Block.TYPE['BLOCKGROUP'];
497 
498 	if (chesterGL.webglMode) {
499 		if (!inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
500 			gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
501 		}
502 		if (this.isFrameDirty || (inBlockGroup && this.isTransformDirty)) {
503 			// NOTE
504 			// the tex coords and the frame coords need to match. Otherwise you get a distorted image
505 			var offset = 9;
506 			var hw = this.contentSize.width * 0.5, hh = this.contentSize.height * 0.5;
507 			var _idx = this.baseBufferIndex * chesterGL.Block.BUFFER_SIZE;
508 			var z = this.position[2];
509 
510 			// NOTE
511 			// this is going to be slow :P
512 			// this is using the matrix to transform the vertex data
513 			if (inBlockGroup) {
514 				var tr = [ hw,  hh, 0],
515 					tl = [-hw,  hh, 0],
516 					br = [ hw, -hh, 0],
517 					bl = [-hw, -hh, 0];
518 				goog.vec.Mat4.multVec3(this.mvMatrix, tr, tr);
519 				goog.vec.Mat4.multVec3(this.mvMatrix, tl, tl);
520 				goog.vec.Mat4.multVec3(this.mvMatrix, bl, bl);
521 				goog.vec.Mat4.multVec3(this.mvMatrix, br, br);
522 
523 				bufferData[_idx           ] =  bl[0]; bufferData[_idx + 1           ] =  bl[1]; bufferData[_idx + 2           ] = z;
524 				bufferData[_idx +   offset] =  tl[0]; bufferData[_idx + 1 +   offset] =  tl[1]; bufferData[_idx + 2 +   offset] = z;
525 				bufferData[_idx + 2*offset] =  br[0]; bufferData[_idx + 1 + 2*offset] =  br[1]; bufferData[_idx + 2 + 2*offset] = z;
526 				bufferData[_idx + 3*offset] =  tr[0]; bufferData[_idx + 1 + 3*offset] =  tr[1]; bufferData[_idx + 2 + 3*offset] = z;
527 			} else {
528 				bufferData[_idx           ] = -hw; bufferData[_idx + 1           ] = -hh; bufferData[_idx + 2           ] = 0;
529 				bufferData[_idx +   offset] = -hw; bufferData[_idx + 1 +   offset] =  hh; bufferData[_idx + 2 +   offset] = 0;
530 				bufferData[_idx + 2*offset] =  hw; bufferData[_idx + 1 + 2*offset] = -hh; bufferData[_idx + 2 + 2*offset] = 0;
531 				bufferData[_idx + 3*offset] =  hw; bufferData[_idx + 1 + 3*offset] =  hh; bufferData[_idx + 2 + 3*offset] = 0;
532 			}
533 
534 			if (this.program == chesterGL.Block.PROGRAM['TEXTURE']) {
535 				var tex = chesterGL.getAsset("texture", this.texture);
536 				var texW = tex.width,
537 					texH = tex.height;
538 				var l = this.frame[0] / texW,
539 					b = this.frame[1] / texH,
540 					w = this.frame[2] / texW,
541 					h = this.frame[3] / texH;
542 				_idx += 3;
543 				bufferData[_idx           ] = l  ; bufferData[_idx+1           ] = b;
544 				bufferData[_idx +   offset] = l  ; bufferData[_idx+1 +   offset] = b+h;
545 				bufferData[_idx + 2*offset] = l+w; bufferData[_idx+1 + 2*offset] = b;
546 				bufferData[_idx + 3*offset] = l+w; bufferData[_idx+1 + 3*offset] = b+h;
547 			}
548 		}
549 		if (this.isColorDirty) {
550 			_idx = 5 + this.baseBufferIndex * chesterGL.Block.BUFFER_SIZE;
551 			var color = this.color;
552 			var opacity = this.opacity;
553 			for (var i=0; i < 4; i++) {
554 				bufferData[_idx     + offset*i] = color[0] * opacity;
555 				bufferData[_idx + 1 + offset*i] = color[1] * opacity;
556 				bufferData[_idx + 2 + offset*i] = color[2] * opacity;
557 				bufferData[_idx + 3 + offset*i] = color[3] * opacity;
558 			}
559 		}
560 		if (chesterGL.webglMode && !inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
561 			gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
562 		}
563 	}
564 }

566 /**
567  * prepares the block for the rendering (transforms if necessary)
568  * @ignore
569  */
570 chesterGL.Block.prototype.visit = function () {
571 	this._inVisit = true;
572 	if (this.update) {
573 		this.update(chesterGL.delta);
574 	}
575 	if (!this.visible) {
576 		this._inVisit = false;
577 		return;
578 	}
579 	this.transform();
580 
581 	var children = this.children;
582 	var len = children.length;
583 	for (var i=0; i < len; i++) {
584 		children[i].visit();
585 	}
586 
587 	// render this block if not in a block group
588 	if (!this.parent || this.parent.type != chesterGL.Block.TYPE['BLOCKGROUP']) {
589 		this.render();
590 	}
591 	// reset our dirty markers
592 	this.isFrameDirty = this.isColorDirty = this.isTransformDirty = false;
593 	this._inVisit = false;
594 
595 	// do we have blocks scheduled to be removed/added?
596 	var b;
597 	while (b = this._scheduledAdd.shift()) {
598 		this.addChild(b);
599 	}
600 	while (b = this._scheduledRemove.shift()) {
601 		this.removeChild(b);
602 	}
603 }

605 /**
606  * render (only will work for non-blockgroup blocks)
607  * @ignore
608  */
609 chesterGL.Block.prototype.render = function () {
610 	if (this.type == chesterGL.Block.TYPE['BLOCKGROUP']) {
611 		throw "Cannot call render on a BlockGroup block!";
612 	}
613 	// dummy render for scene blocks
614 	if (this.type == chesterGL.Block.TYPE['SCENE']) {
615 		return;
616 	}
617 
618 	if (chesterGL.webglMode) {
619 		var gl = chesterGL.gl;
620 		// select current shader
621 		var program = chesterGL.selectProgram(chesterGL.Block.PROGRAM_NAME[this.program]);
622 
623 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
624 		var texOff = 3 * 4,
625 			colorOff = texOff + 2 * 4,
626 			stride = chesterGL.Block.QUAD_SIZE;
627 
628 		gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
629 		gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, colorOff);
630 
631 		if (this.program == chesterGL.Block.PROGRAM['DEFAULT']) {
632 			// no extra attributes for the shader
633 		} else if (this.program == chesterGL.Block.PROGRAM['TEXTURE']) {
634 			var texture = chesterGL.getAsset('texture', this.texture);
635 
636 			// pass the texture attributes
637 			gl.vertexAttribPointer(program.attribs['textureCoordAttribute'], 2, gl.FLOAT, false, stride, texOff);
638 
639 			gl.activeTexture(gl.TEXTURE0);
640 			gl.bindTexture(gl.TEXTURE_2D, texture.tex);
641 			gl.uniform1i(program.samplerUniform, 0);
642 			gl.activatedTexture = true;
643 		}
644 
645 		// set the matrix uniform (the multiplied model view projection matrix)
646 		var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
647 		if (transformDirty) {
648 			goog.vec.Mat4.multMat(chesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
649 		}
650 		gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
651 		gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
652 	} else {
653 		var gl = chesterGL.offContext;
654 		// canvas drawing api - we only draw textures
655 		if (this.program == chesterGL.Block.PROGRAM.TEXTURE) {
656 			var m = this.mvMatrix;
657 			var texture = chesterGL.getAsset('texture', this.texture);
658 			gl.globalAlpha = this.opacity;
659 			gl.setTransform(m[0], m[1], m[4], m[5], m[12], gl.viewportHeight - m[13]);
660 			var w = this.contentSize.width, h = this.contentSize.height;
661 			var frame = this.frame;
662 			gl.drawImage(texture, frame[0], texture.height - (frame[1] + h), frame[2], frame[3], -w/2, -h/2, w, h);
663 		}
664 	}
