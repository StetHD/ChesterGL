 26 (function (window) {
 27 	var ChesterGL = window['ChesterGL'];
 28 	
 29 	/**
 30 	 * @constructor
 31 	 * @param {ChesterGL.Block} block
 32 	 * @param {number=} totalTime
 33 	 * @return ChesterGL.Action
 34 	 */
 35 	ChesterGL.Action = function (block, totalTime) {
 36 		this.block = block;
 37 		this.totalTime = totalTime * 1000;
 38 		this.elapsed = 0;
 39 	};
 40 	
 41 	/**
 42 	 * The block to which this action will be applied
 43 	 * 
 44 	 * @type {?ChesterGL.Block}
 45 	 */
 46 	ChesterGL.Action.prototype.block = null;
 47 	
 48 	/**
 49 	 * The total time in seconds this action should take
 50 	 * (might not be relevant for all actions)
 51 	 * 
 52 	 * @type {number}
 53 	 */
 54 	ChesterGL.Action.prototype.totalTime = 0;
 55 
 56 	/**
 57 	 * Current time in seconds of the action
 58 	 * (might not be relevant for all actions)
 59 	 * 
 60 	 * @type {number}
 61 	 */
 62 	ChesterGL.Action.prototype.elapsed = 0;
 63 	
 64 	/**
 65 	 * The current time of the action
 66 	 * @type {number}
 67 	 */
 68 	ChesterGL.Action.prototype.currentTime = 0;
 69 
 70 	/**
 71 	 * The current time of the action
 72 	 * @type {boolean}
 73 	 */
 74 	ChesterGL.Action.prototype.finished = false;
 75 	
 76 	/**
 77 	 * This is the default delta function (does nothing)
 78 	 * @param {number} delta
 79 	 */
 80 	ChesterGL.Action.prototype.update = function (delta) {
 81 		this.elapsed += delta;
 82 		if (this.elapsed >= this.totalTime) {
 83 			this.finished = true;
 84 		}
 85 	};
 86 	
 87 	/**
 88 	 * @constructor
 89 	 * @extends ChesterGL.Action
 90 	 * @param {ChesterGL.Block} block
 91 	 * @param {number} totalTime
 92 	 * @param {vec3} finalPosition
 93 	 * @return ChesterGL.MoveAction
 94 	 */
 95 	ChesterGL.MoveAction = function (block, totalTime, finalPosition) {
 96 		ChesterGL.Action.call(this, block, totalTime);
 97 		this.finalPosition = finalPosition;
 98 		this.startPosition = vec3.create(block.position);
 99 		this.deltaPosition = vec3.create();
100 		vec3.subtract(this.finalPosition, this.startPosition, this.deltaPosition);
101 	};
102 	
103 	/**
104 	 * extend from Action
105 	 * @ignore
106 	 */
107 	ChesterGL.MoveAction.prototype = Object.create(ChesterGL.Action.prototype);
108 	
109 	/**
110 	 * @type {vec3}
111 	 */
112 	ChesterGL.MoveAction.prototype.finalPosition = null;
113 	
114 	/**
115 	 * @param {number} delta miliseconds from last time we updated
116 	 */
117 	ChesterGL.MoveAction.prototype.update = function (delta) {
118 		ChesterGL.Action.prototype.update.call(this, delta);
119 		if (this.finished) {
120 			this.block.moveTo(this.finalPosition);
121 		} else {		
122 			var t = Math.min(1, this.elapsed / this.totalTime);
123 			var st = this.startPosition;
124 			var d = this.deltaPosition;
125 			var dx = d[0] * t;
126 			// console.log("t: " + t + "\t(" + dx + ")");
127 			this.block.moveTo([st[0] + d[0] * t, st[1] + d[1] * t, st[2] + d[2] * t]);		
128 		}
129 	}
130 		
131 	/**
132 	 * global action manager
133 	 * @namespace
134 	 */
135 	ChesterGL.ActionManager = {};
136 	
137 	
138 	/**
139 	 * the list of scheduled actions
140 	 * @ignore
141 	 * @type {Array.<ChesterGL.Action>}
142 	 */
143 	ChesterGL.ActionManager.scheduledActions_ = [];
144 	
145 	/**
146 	 * adds an action to the scheduler
147 	 * 
148 	 * @param {ChesterGL.Action} action
149 	 */
150 	ChesterGL.ActionManager.scheduleAction = function (action) {
151 		this.scheduledActions_.push(action);
152 	}
153 	
154 	/**
155 	 * Iterate over all scheduled actions
156 	 * @param {number} delta number of miliseconds to run in all actions
157 	 */
158 	ChesterGL.ActionManager.tick = function (delta) {
159 		var i = 0, len = this.scheduledActions_.length;
160 		for (i=0; i < len; i++) {
161 			var a = this.scheduledActions_[i];
162 			!a.finished && a.update(delta);
163 		}
164 	}
165 })(window);
