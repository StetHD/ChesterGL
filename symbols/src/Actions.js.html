 26 (function (window) {
 27 	"use strict";
 28 
 29 	var ChesterGL = window['ChesterGL'];
 30 
 31 	/**
 32 	 * @constructor
 33 	 * @param {ChesterGL.Block} block
 34 	 * @param {number=} totalTime
 35 	 * @return ChesterGL.Action
 36 	 */
 37 	ChesterGL.Action = function (block, totalTime) {
 38 		this.block = block;
 39 		this.totalTime = totalTime * 1000;
 40 		this.elapsed = 0;
 41 	};
 42 	
 43 	/**
 44 	 * The block to which this action will be applied
 45 	 * 
 46 	 * @type {?ChesterGL.Block}
 47 	 */
 48 	ChesterGL.Action.prototype.block = null;
 49 	
 50 	/**
 51 	 * The total time in seconds this action should take
 52 	 * (might not be relevant for all actions)
 53 	 * 
 54 	 * @type {number}
 55 	 */
 56 	ChesterGL.Action.prototype.totalTime = 0;
 57 
 58 	/**
 59 	 * Current time in seconds of the action
 60 	 * (might not be relevant for all actions)
 61 	 * 
 62 	 * @type {number}
 63 	 */
 64 	ChesterGL.Action.prototype.elapsed = 0;
 65 	
 66 	/**
 67 	 * The current time of the action
 68 	 * @type {number}
 69 	 */
 70 	ChesterGL.Action.prototype.currentTime = 0;
 71 
 72 	/**
 73 	 * The current time of the action
 74 	 * @type {boolean}
 75 	 */
 76 	ChesterGL.Action.prototype.finished = false;
 77 	
 78 	/**
 79 	 * This is the default delta function (does nothing)
 80 	 * @param {number} delta
 81 	 */
 82 	ChesterGL.Action.prototype.update = function (delta) {
 83 		this.elapsed += delta;
 84 		if (this.totalTime > 0 && this.elapsed >= this.totalTime) {
 85 			this.finished = true;
 86 		}
 87 	};
 88 	
 89 	/**
 90 	 * @constructor
 91 	 * @extends ChesterGL.Action
 92 	 * @param {ChesterGL.Block} block
 93 	 * @param {number} totalTime
 94 	 * @param {vec3} finalPosition
 95 	 * @return ChesterGL.MoveAction
 96 	 */
 97 	ChesterGL.MoveToAction = function (block, totalTime, finalPosition) {
 98 		ChesterGL.Action.call(this, block, totalTime);
 99 		this.finalPosition = vec3.create(finalPosition);
100 		this.startPosition = vec3.create(block.position);
101 	};
102 	
103 	/**
104 	 * extend from Action
105 	 * @ignore
106 	 */
107 	ChesterGL.MoveToAction.prototype = Object.create(ChesterGL.Action.prototype);
108 		
109 	/**
110 	 * @type {?Object}
111 	 */
112 	ChesterGL.MoveToAction.prototype.finalPosition = null;
113 
114 	/**
115 	 * @type {?Object}
116 	 */
117 	ChesterGL.MoveToAction.prototype.startPosition = null;
118 	
119 	/**
120 	 * @type {vec3}
121 	 * @ignore
122 	 */
123 	var __tmp_pos = vec3.create();
124 
125 	/**
126 	 * @param {number} delta miliseconds from last time we updated
127 	 */
128 	ChesterGL.MoveToAction.prototype.update = function (delta) {
129 		ChesterGL.Action.prototype.update.call(this, delta);
130 		var block = this.block;
131 		if (this.finished) {
132 			block.moveTo(this.finalPosition);
133 		} else {		
134 			var t = Math.min(1, this.elapsed / this.totalTime);
135 			// console.log("t: " + t + "\t(" + dx + ")");
136 			vec3.lerp(this.startPosition, this.finalPosition, t, __tmp_pos);
137 			block.moveTo(__tmp_pos);
138 		}
139 	};
140 
141 	/**
142 	 * @constructor
143 	 * @extends ChesterGL.Action
144 	 * @param {ChesterGL.Block} block
145 	 * @param {number} delay in seconds
146 	 * @param {Array.<Object>} frames
147 	 * @param {boolean?} loop
148 	 * @return ChesterGL.AnimateAction
149 	 */
150 	ChesterGL.AnimateAction = function (block, delay, frames, loop) {
151 		this.delay = delay * 1000.0;
152 		var totalTime = this.delay * frames.length;
153 		if (loop == true) totalTime = -1;
154 		ChesterGL.Action.call(this, block, totalTime);
155 		this.shouldLoop = (loop == true);
156 		this.frames = frames.slice(0);
157 	};
158 
159 	/**
160 	 * extend from Action
161 	 * @ignore
162 	 */
163 	ChesterGL.AnimateAction.prototype = Object.create(ChesterGL.Action.prototype);
164 
165 	/**
166 	 * the current frame
167 	 * @type {number}
168 	 */
169 	ChesterGL.AnimateAction.prototype.currentFrame = 0;
170 
171 	/**
172 	 * The delay between frames
173 	 * @type {number}
174 	 */
175 	ChesterGL.AnimateAction.prototype.delay = 0.0;
176 
177 	/**
178 	 * The total frames of the animation
179 	 * @type {Array.<Object>}
180 	 */
181 	ChesterGL.AnimateAction.prototype.frames = null;
182 
183 	/**
184 	 * Whether or not the animation should loop
185 	 * @type {boolean}
186 	 */
187 	ChesterGL.AnimateAction.prototype.shouldLoop = false;
188 
189 	/**
190 	 * @param {number} delta
191 	 */
192 	ChesterGL.AnimateAction.prototype.update = function (delta) {
193 		ChesterGL.Action.prototype.update.call(this, delta);
194 		var block = this.block;
195 		if (this.finished) {
196 			this.currentFrame = this.frames.length - 1;
197 			this.finished = true;
198 			// set last frame
199 			block.setFrame(this.frames[this.currentFrame]);
200 		} else {
201 			if (this.elapsed >= this.delay * this.currentFrame) {
202 				block.setFrame(this.frames[this.currentFrame++]);
203 				if (this.currentFrame == this.frames.length) {
204 					if (this.shouldLoop) { this.currentFrame = 0; this.elapsed = 0; }
205 					else this.finished = true;
206 				}
207 			}
208 		}
209 	};
210 
211 	/**
212 	 * global action manager
213 	 * @namespace
214 	 * @type {Object.<string,Object>}
215 	 */
216 	ChesterGL.ActionManager = {};
217 
218 	/**
219 	 * the list of scheduled actions
220 	 * @ignore
221 	 * @type {Array.<ChesterGL.Action>}
222 	 */
223 	ChesterGL.ActionManager.scheduledActions_ = [];
224 
225 	/**
226 	 * adds an action to the scheduler
227 	 * 
228 	 * @param {ChesterGL.Action} action
229 	 */
230 	ChesterGL.ActionManager.scheduleAction = function (action) {
231 		this.scheduledActions_.push(action);
232 	}
233 
234 	/**
235 	 * Iterate over all scheduled actions
236 	 * @param {number} delta number of miliseconds to run in all actions
237 	 */
238 	ChesterGL.ActionManager.tick = function (delta) {
239 		var i = 0, len = this.scheduledActions_.length;
240 		for (i=0; i < len; i++) {
241 			var a = this.scheduledActions_[i];
242 			!a.finished && a.update(delta);
243 		}
244 	}
245 
246 	ChesterGL.exportProperty(ChesterGL, 'ActionManager', ChesterGL.ActionManager);
247 	ChesterGL.exportProperty(ChesterGL.ActionManager, 'scheduleAction', ChesterGL.ActionManager.scheduleAction);
248 	ChesterGL.exportProperty(ChesterGL.ActionManager, 'tick', ChesterGL.ActionManager.tick);
249 	ChesterGL.exportProperty(ChesterGL, 'MoveToAction', ChesterGL.MoveToAction);
250 	ChesterGL.exportProperty(ChesterGL, 'AnimateAction', ChesterGL.AnimateAction);
251 })(window);
