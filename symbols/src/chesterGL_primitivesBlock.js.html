 32 /**
 33  * creates a new PrimitiveBlock. All draw functions should be placed in the update
 34  * method of the primitive block. Every frame, all draw calls are reset.
 35  * @constructor
 36  * @param {number=} maxPoints the max number of points, by default it's set to 500
 37  * @param {number=} maxLines the max number of lines, by default it's set to 500
 38  * @extends {chesterGL.Block}
 39  * @example
 40  * var block = new PrimitiveBlock();
 41  * block.setUpdate(function () {
 42  * 	var sz = this.contentSize;
 43  * 	this.drawPoint(sz.width * 0.5, sz.height * 0.5);
 44  * });
 45  */
 46 chesterGL.PrimitiveBlock = function (maxPoints, maxLines) {
 47 	if (!chesterGL.webglMode) {
 48 		throw "PrimitiveBlock only works on WebGL mode";
 49 	}
 50 	this.maxPoints = maxPoints || 500;
 51 	this.maxLines = maxLines || 500;
 52 	chesterGL.Block.call(this, null, chesterGL.Block.TYPE['PRIMITIVE']);
 53 	var gl = chesterGL.gl;
 54 	this.glBufferPoints = gl.createBuffer();
 55 	this.glBufferPointsData = new Float32Array(this.maxPoints * 7);
 56 	this.glBufferLines = gl.createBuffer();
 57 	this.glBufferLinesData = new Float32Array(this.maxLines * 14);
 58 	this.program = chesterGL.Block.PROGRAM['DEFAULT'];
 59 };
 60 goog.inherits(chesterGL.PrimitiveBlock, chesterGL.Block);

120 /**
121  * draws a point of a specified color (white by default)
122  * @param {number} x
123  * @param {number} y
124  * @param {Array|Float32Array=} color An array containing the RGBA color, in floats from 0 to 1
125  */
126 chesterGL.PrimitiveBlock.prototype.drawPoint = function (x, y, color) {
127 	if (this.currentIndexPoint < this.maxPoints) {
128 		var idx = this.currentIndexPoint * 7;
129 		color = color || [1, 1, 1, 1];
130 		this.glBufferPointsData[idx + 0] = x;
131 		this.glBufferPointsData[idx + 1] = y;
132 		this.glBufferPointsData[idx + 2] = 0;
133 		this.glBufferPointsData[idx + 3] = color[0];
134 		this.glBufferPointsData[idx + 4] = color[1];
135 		this.glBufferPointsData[idx + 5] = color[2];
136 		this.glBufferPointsData[idx + 6] = color[3];
137 		this.currentIndexPoint++;
138 	} else {
139 		throw "too many points!";
140 	}
141 };
142 
143 /**
144  * draws a line between (x0, y0) and (x1, y1), of the specified color (defaults to white)
145  * @param {number} x0
146  * @param {number} y0
147  * @param {number} x1
148  * @param {number} y1
149  * @param {Array|Float32Array=} color An array containing the RGBA color, in floats from 0 to 1
150  */
151 chesterGL.PrimitiveBlock.prototype.drawLine = function (x0, y0, x1, y1, color) {
152 	if (this.currentIndexLine < this.maxLines) {
153 		var idx = this.currentIndexLine * 14;
154 		color = color || [1, 1, 1, 1];
155 		this.glBufferLinesData[idx +  0] = x0;
156 		this.glBufferLinesData[idx +  1] = y0;
157 		this.glBufferLinesData[idx +  2] = 0;
158 		this.glBufferLinesData[idx +  3] = color[0];
159 		this.glBufferLinesData[idx +  4] = color[1];
160 		this.glBufferLinesData[idx +  5] = color[2];
161 		this.glBufferLinesData[idx +  6] = color[3];
162 		this.glBufferLinesData[idx +  7] = x1;
163 		this.glBufferLinesData[idx +  8] = y1;
164 		this.glBufferLinesData[idx +  9] = 0;
165 		this.glBufferLinesData[idx + 10] = color[0];
166 		this.glBufferLinesData[idx + 11] = color[1];
167 		this.glBufferLinesData[idx + 12] = color[2];
168 		this.glBufferLinesData[idx + 13] = color[3];
169 		this.currentIndexLine++;
170 	} else {
171 		throw "too many lines!";
172 	}
173 };

175 /**
176  * draws a polygon given an array of points
177  * @param {Array} points The array of points. A point is represented by an array of 3 elements: x, y and z.
178  * @param {Array|Float32Array=} color An array containing the RGBA color, in floats from 0 to 1
179  * @param {boolean=} closePolygon should the polygon be closed? (link the last element with the first). Defaults to false
180  * @param {boolean=} fillPolygon set to true to fill the polygon
181  * @example
182  * // will draw a red 50px wide square on (0, 0)
183  * var points = [[0, 0, 0], [0, 50, 0], [50, 50, 0], [50, 0, 0]];
184  * primitive.drawPolygon(points, [1, 0, 0, 1], true);
185  */
186 chesterGL.PrimitiveBlock.prototype.drawPolygon = function (points, color, closePolygon, fillPolygon) {
187 	color = color || [1, 1, 1, 1];
188 	closePolygon = closePolygon || false;
189 	fillPolygon = fillPolygon || false;
190 	var len = points.length;
191 	var gl = chesterGL.gl;
192 	var buffer = new Float32Array(points.length * 7);
193 	var glbuffer = gl.createBuffer();
194 	for (var i=0; i < len; i++) {
195 		var pt = points[i];
196 		buffer[i*7 + 0] = pt[0];
197 		buffer[i*7 + 1] = pt[1];
198 		buffer[i*7 + 2] = pt[2];
199 		buffer[i*7 + 3] = color[0];
200 		buffer[i*7 + 4] = color[1];
201 		buffer[i*7 + 5] = color[2];
202 		buffer[i*7 + 6] = color[3];
203 	}
204 	gl.bindBuffer(gl.ARRAY_BUFFER, glbuffer);
205 	gl.bufferData(gl.ARRAY_BUFFER, buffer, gl.STATIC_DRAW);
206 	this.polygons.unshift([buffer, glbuffer, closePolygon, fillPolygon]);
207 };

209 /**
210  * recreates the point buffer
211  * @ignore
212  */
213 chesterGL.PrimitiveBlock.prototype.recreatePointBuffer = function () {
214 	var gl = chesterGL.gl;
215 	var endIndex = (this.currentIndexPoint) * 7;
216 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBufferPoints);
217 	gl.bufferData(gl.ARRAY_BUFFER, this.glBufferPointsData.subarray(0, endIndex), gl.STATIC_DRAW);
218 };

220 /**
221  * recreates the line buffer
222  * @ignore
223  */
224 chesterGL.PrimitiveBlock.prototype.recreateLineBuffer = function () {
225 	var gl = chesterGL.gl;
226 	var endIndex = (this.currentIndexLine) * 14;
227 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBufferLines);
228 	gl.bufferData(gl.ARRAY_BUFFER, this.glBufferLinesData.subarray(0, endIndex), gl.STATIC_DRAW);
229 }

231 /**
232  * @ignore
233  */
234 chesterGL.PrimitiveBlock.prototype.visit = function () {
235 	this.currentIndexPoint = 0;
236 	this.currentIndexLine = 0;
237 	// only reset if we need to
238 	if (this.polygons.length > 0)
239 		this.polygons = [];
240 	chesterGL.Block.prototype.visit.call(this);
241 };

243 /**
244  * the actual render function
245  * @ignore
246  */
247 chesterGL.PrimitiveBlock.prototype.render = function () {
248 	var gl = chesterGL.gl;
249 	var program = chesterGL.selectProgram(chesterGL.Block.PROGRAM_NAME[this.program]);
250 	
251 	if (this.currentIndexPoint > 0 || this.currentIndexLine > 0 || this.polygons.length > 0) {
252 		// set the matrix uniform (the multiplied model view projection matrix)
253 		goog.vec.Mat4.multMat(chesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
254 		gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
255 	}
256 	var pointSize = 3 + 4;
257 	var stride = pointSize * 4;
258 
259 	// render all points
260 	if (this.currentIndexPoint > 0) {
261 		this.recreatePointBuffer();
262 
263 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBufferPoints);
264 		gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
265 		gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, 12);
266 
267 		// draw the points
268 		gl.drawArrays(gl.POINTS, 0, this.currentIndexPoint);
269 	}
270 
271 	// render all lines
272 	if (this.currentIndexLine > 0) {
273 		this.recreateLineBuffer();
274 
275 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBufferLines);
276 		gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
277 		gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, 12);
278 
279 		// draw the lines
280 		gl.drawArrays(gl.LINES, 0, this.currentIndexLine * 2);
281 	}
282 
283 	// render all polygons
284 	var len = this.polygons.length;
285 	if (len > 0) {
286 		for (var i=0; i < len; i++) {
287 			var poly = this.polygons[i];
288 			gl.bindBuffer(gl.ARRAY_BUFFER, poly[1]);
289 			gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
290 			gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, 12);
291 			if (poly[2]) {
292 				gl.drawArrays(gl.LINE_LOOP, 0, poly[0].length / 7);
293 			} else {
294 				gl.drawArrays(gl.LINE_STRIP, 0, poly[0].length / 7);
295 			}
296 		}
297 	}
298 };
