 35 	/**
 36 	 * BUFFER_ELEMENTS * 4 (bytes) ~> BUFFER_ELEMENTS floats == 1 (lifetime) + 1 (start time) + 1 (start size) + 1 (end size) + 3 (start pos) + 3 (end pos)
 37 	 * @ignore

 42 	/**
 43 	 * @ignore
 44 	 * @param {vec3} original The original vector
 45 	 * @param {vec3} variance the variance for every coordinate in the original vector
 46 	 * @return {vec3}
 47 	 */
 48 	function randomVec3(original, variance) {
 49 		var vec = vec3.create();
 50 		if (variance) {
 51 			vec[0] = original[0] + (variance[0] * 2 * Math.random());
 52 			vec[1] = original[1] + (variance[1] * 2 * Math.random());
 53 			vec[2] = original[2] + (variance[2] * 2 * Math.random());
 54 		} else {
 55 			vec[0] = original[0];
 56 			vec[1] = original[1];
 57 			vec[2] = original[2];
 58 		}
 59 		return vec;
 60 	}

 68 	ChesterGL.ParticleSystem = function (properties) {
 69 		ChesterGL.Block.call(this, null, ChesterGL.Block.TYPE.PARTICLE);

 71 		ChesterGL.loadAsset('texture', properties['texture'], function () {
 72 			_this.loadProperties(properties);

 83 	 * Load the shaders for the particle system
 84 	 */
 85 	ChesterGL.ParticleSystem.loadShaders = function () {
 86 		ChesterGL.initShader("particles", function (program) {
 87 			var gl = ChesterGL.gl;
 88 			program.mvpMatrixUniform = gl.getUniformLocation(program, "uMVPMatrix");
 89 			program.uSampler         = gl.getUniformLocation(program, "uSampler");
 90 			program.u_time           = gl.getUniformLocation(program, "u_time");
 91 			program.u_startColor     = gl.getUniformLocation(program, "u_startColor");
 92 			program.u_endColor       = gl.getUniformLocation(program, "u_endColor");
 93 			program.attribs = {
 94 				'a_startPosition': gl.getAttribLocation(program, 'a_startPosition'),
 95 				'a_lifetime'  : gl.getAttribLocation(program, 'a_lifetime'),
 96 				'a_startTime'  : gl.getAttribLocation(program, 'a_startTime'),
 97 				'a_startSize'  : gl.getAttribLocation(program, 'a_startSize'),
 98 				'a_endSize'  : gl.getAttribLocation(program, 'a_endSize'),
 99 				'a_speed'        : gl.getAttribLocation(program, 'a_speed')
100 			};
101 						
102 			// test for errors on gl
103 			var error = gl.getError();
104 			if (error != 0) {
105 				console.log("gl error: " + error);
106 			}
107 		});
108 		ChesterGL.ParticleSystem._shadersLoaded = true;
109 	}

212 	/**
213 	 * @param {Object} properties
214 	 */
215 	ChesterGL.ParticleSystem.prototype.loadProperties = function (properties) {
216 		this.program = -1;
217 		if (!ChesterGL.ParticleSystem._shadersLoaded) {
218 			ChesterGL.ParticleSystem.loadShaders();
219 		}
220 		this.particleTexture = properties['texture'];
221 		this.maxParticles = properties['maxParticles'];
222 		this.duration = parseFloat(properties['duration']) * 1000.0;
223 		this.lifetime = parseFloat(properties['lifetime']) * 1000.0;
224 		this.lifetimeVariance = parseFloat(properties['lifetimeVariance']) * 1000.0;
225 		this.startColor = quat4.create(properties['startColor']);
226 		this.positionVariance = vec3.create(properties['positionVariance']);
227 		this.endColor = quat4.create(properties['endColor']);
228 		this.particleSpeed = vec3.create(properties['speed']);
229 		this.particleSpeedVariance = vec3.create(properties['speedVariance']);
230 		this.elapsedTime = 0;
231 		this.blendOptions = properties['blendOptions'].slice(0); // copy the array
232 		
233 		this.glBuffer = ChesterGL.gl.createBuffer();
234 		this.glBufferData = new Float32Array(this.maxParticles * BUFFER_ELEMENTS);
235 		this.resetParticles();
236 	}

238 	/**
239 	 * adds a new particle (sets the lifetime in the data sent to the shader)
240 	 */
241 	ChesterGL.ParticleSystem.prototype.addParticle = function () {
242 		var lifespan = Math.abs(this.lifetime + this.lifetimeVariance * (Math.random() * 2 - 1));
243 		this.initParticle(this.particleCount, lifespan, this.elapsedTime);
244 		this.particleCount++;
245 		this.particleAdded = true;
246 	}
247 	
248 	/**
249 	 * @param {number} idx
250 	 * @param {number=} lifetime
251 	 * @param {number=} startTime
252 	 */
253 	ChesterGL.ParticleSystem.prototype.initParticle = function (idx, lifetime, startTime) {
254 		var d = this.glBufferData;
255 		lifetime = lifetime || -1.0;
256 		startTime = startTime || 0.0;
257 		
258 		// lifetime, start time, start size, end size
259 		d[idx * BUFFER_ELEMENTS + 0] = lifetime;
260 		d[idx * BUFFER_ELEMENTS + 1] = startTime;
261 		d[idx * BUFFER_ELEMENTS + 2] = 40.0;
262 		d[idx * BUFFER_ELEMENTS + 3] = 1.0;
263 		
264 		// speed
265 		d[idx * BUFFER_ELEMENTS + 4] = this.particleSpeed[0] + this.particleSpeedVariance[0] * (Math.random() * 2 - 1);
266 		d[idx * BUFFER_ELEMENTS + 5] = this.particleSpeed[1] + this.particleSpeedVariance[1] * (Math.random() * 2 - 1);
267 		d[idx * BUFFER_ELEMENTS + 6] = this.particleSpeed[2] + this.particleSpeedVariance[2] * (Math.random() * 2 - 1);
268 		
269 		// start position
270 		d[idx * BUFFER_ELEMENTS + 7] = (Math.random() * 2 - 1) * this.positionVariance[0];
271 		d[idx * BUFFER_ELEMENTS + 8] = (Math.random() * 2 - 1) * this.positionVariance[1];
272 		d[idx * BUFFER_ELEMENTS + 9] = (Math.random() * 2 - 1) * this.positionVariance[2];
273 	}

275 	/**
276 	 * reset particle data - this is slow!
277 	 */
278 	ChesterGL.ParticleSystem.prototype.resetParticles = function () {
279 		var program = ChesterGL.selectProgram("particles");
280 		var gl = ChesterGL.gl;
281 		for (var i = 0; i < this.maxParticles; i++) {
282 			this.initParticle(i);
283 		}
284 		gl.uniform4fv(program.u_startColor, this.startColor);
285 		gl.uniform4fv(program.u_endColor  , this.endColor);
286 		gl.uniform1i(program.uSampler, 0);
287 		
288 		this.sendParticleData(program);
289 		
290 		this.particleCount = this.emissionCounter = 0;
291 		// how many particles are emitted per second
292 		this.emissionRate = this.maxParticles / Math.abs(this.lifetime);
293 	}
294 	
295 	/**
296 	 * will send the particle data to the gpu
297 	 * @program {WebGLProgram} program
298 	 */
299 	ChesterGL.ParticleSystem.prototype.sendParticleData = function (program) {
300 		var gl = ChesterGL.gl;
301 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
302 		gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
303 	}

305 	var _ps_tmp = new Float32Array(BUFFER_ELEMENTS);
306 	
307 	ChesterGL.ParticleSystem.prototype.update = function (delta) {
308 		var program = ChesterGL.selectProgram("particles");
309 		if (!program) {
310 			return;
311 		}
312 		this.elapsedTime += delta;
313 		
314 		// how many seconds until the next particle
315 		var rate = 1.0 / this.emissionRate;
316 		this.emissionCounter += delta;
317 		while (this.particleCount < this.maxParticles && this.emissionCounter > rate) {
318 			this.addParticle();
319 			this.emissionCounter -= rate;
320 		}
321 
322 		for (var i = 0; i < this.maxParticles; i++) {
323 			var buffer = this.glBufferData;
324 			var idx = i * BUFFER_ELEMENTS;
325 			// if expired, move the (buffer) particle ahead
326 			if (buffer[idx] > 0 && (buffer[idx] + buffer[idx+1]) <= this.elapsedTime && i != this.particleCount - 1) {
327 				// copy the particle into the tmp buffer and invalidate
328 				var tmp = buffer.subarray(idx, idx + BUFFER_ELEMENTS);
329 				_ps_tmp.set(tmp);
330 				_ps_tmp[0] = -1.0;
331 				// shift the array from idx to particleCount
332 				tmp = buffer.subarray(idx + BUFFER_ELEMENTS, this.particleCount * BUFFER_ELEMENTS);
333 				buffer.set(tmp, idx);
334 				// copy the old particle in the last spot
335 				buffer.set(_ps_tmp, (this.particleCount-1) * BUFFER_ELEMENTS);
336 				// decrease the particle count
337 				this.particleCount --;
338 			}
339 		}
340 
341 		if (this.duration > 0 && this.elapsedTime > this.duration) {
342 			this.elapsedTime = 0;
343 		}
344 	}

346 	ChesterGL.ParticleSystem.prototype.render = function () {
347 		var program = ChesterGL.selectProgram("particles");
348 		if (!program) {
349 			return;
350 		}
351 		var gl = ChesterGL.gl;
352 		var texture = ChesterGL.getAsset('texture', this.particleTexture);
353 		
354 		gl.enable(gl.BLEND);
355 		gl.blendFunc(gl[this.blendOptions[0]], gl[this.blendOptions[1]]);
356 		
357 		if (this.particleAdded) {
358 			this.sendParticleData(program);
359 			this.particleAdded = false;
360 		}
361 		
362 		// send the elapsed time
363 		gl.uniform1f(program.u_time, this.elapsedTime);
364 		
365 		// activate the texture
366 		gl.activeTexture(gl.TEXTURE0);
367 		gl.bindTexture(gl.TEXTURE_2D, texture.tex);
368 		
369 		// bind buffer
370 		gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
371 		gl.vertexAttribPointer(program.attribs['a_lifetime']     , 3, gl.FLOAT, false, PARTICLE_SIZE, 0);
372 		gl.vertexAttribPointer(program.attribs['a_startTime']    , 3, gl.FLOAT, false, PARTICLE_SIZE, 4);
373 		gl.vertexAttribPointer(program.attribs['a_startSize']    , 3, gl.FLOAT, false, PARTICLE_SIZE, 8);
374 		gl.vertexAttribPointer(program.attribs['a_endSize']      , 3, gl.FLOAT, false, PARTICLE_SIZE, 12);
375 		gl.vertexAttribPointer(program.attribs['a_speed']        , 3, gl.FLOAT, false, PARTICLE_SIZE, 16);
376 		gl.vertexAttribPointer(program.attribs['a_startPosition'], 3, gl.FLOAT, false, PARTICLE_SIZE, 28);
377 		
378 		// and draw:
379 		var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
380 		if (transformDirty) {
381 			mat4.multiply(ChesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
382 		}
383 		gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
384 		gl.drawArrays(gl.POINTS, 0, this.maxParticles);
385 	}
