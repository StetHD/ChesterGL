 41 HTMLCanvasElement.prototype.relativePosition = function (evt) {
 42 	var pt = HTMLCanvasElement._canvas_tmp_mouse;
 43 	pt.x = 0; pt.y = 0;
 44 
 45 	var off = $(this).offset(), height = $(this).height();
 46 	pt.x = (evt.pageX - off.left);
 47 	pt.y = (height - (evt.pageY - off.top));
 48 	return pt;
 49 };

252 /**
253  * sets the current program, also sets the uniforms for that shader
254  * 
255  * @param {string} program
256  * @return {WebGLProgram}
257  * @ignore
258  */
259 chesterGL.selectProgram = function (program) {
260 	var prog = chesterGL.programs[program];
261 	var gl = chesterGL.gl;
262 	if (program != chesterGL.currentProgram) {
263 		// console.log("selecting program " + program);
264 		chesterGL.currentProgram = program;
265 		gl.validateProgram(prog);
266 		gl.useProgram(prog);
267 		// enable attribs
268 		for (var attr in prog.attribs) {
269 			// console.log("  enabling attribute " + attr);
270 			gl.enableVertexAttribArray(prog.attribs[attr]);
271 		}
272 	}
273 	return prog;
274 };

276 /**
277  * setups the webgl canvas
278  * @param {string} canvasId
279  */
280 chesterGL.setup = function (canvasId) {
281 	var canvas = document.getElementById(canvasId); // get the DOM element
282 	chesterGL.initGraphics(canvas);
283 	if (chesterGL.webglMode) {
284 		chesterGL.initDefaultShaders();
285 	}
286 	var queryStr = window.location.search.substring(1);
287 	var keyValues = queryStr.split('&');
288 	for (var i in keyValues) {
289 		var key = keyValues[i].split('=');
290 		if (key[0] == '_cdbg' && key[1] == '1') {
291 			chesterGL.debugSprite = true;
292 			console.log("debug mode on");
293 		}
294 	}
295 
296 	chesterGL.debugSpan = document.getElementById("debug-info");
297 	// register the default handler for textures
298 	chesterGL.registerAssetHandler('texture', chesterGL.defaultTextureHandler);
299 	chesterGL.registerAssetLoader('texture', chesterGL.defaultTextureLoader);
300 	chesterGL.registerAssetLoader('default', chesterGL.defaultAssetLoader);
301 };

303 /**
304  * tryies to init the graphics stuff:
305  * 1st attempt: webgl
306  * fallback: canvas
307  * 
308  * @param {Element} canvas
309  */
310 chesterGL.initGraphics = function (canvas) {
311 	try {
312 		chesterGL.canvas = canvas;
313 		if (chesterGL.webglMode) {
314 			chesterGL.gl = canvas.getContext("experimental-webgl", {alpha: false, antialias: false});
315 			if (chesterGL.gl && window['WebGLDebugUtils']) {
316 				console.log("installing debug context");
317 				chesterGL.gl = WebGLDebugUtils.makeDebugContext(chesterGL.gl, throwOnGLError);
318 			}
319 		}
320 	} catch (e) {
321 		console.log("ERROR: " + e);
322 	}
323 	if (!chesterGL.gl) {
324 		// fallback to canvas API (uses an offscreen buffer)
325 		chesterGL.gl = canvas.getContext("2d");
326 		if (chesterGL.usesOffscreenBuffer) {
327 			chesterGL.offCanvas = document.createElement('canvas');
328 			chesterGL.offCanvas.width = canvas.width;
329 			chesterGL.offCanvas.height = canvas.height;
330 			chesterGL.offContext = chesterGL.offCanvas.getContext("2d");
331 			chesterGL.offContext.viewportWidth = canvas.width;
332 			chesterGL.offContext.viewportHeight = canvas.height;
333 		} else {
334 			chesterGL.offContext = chesterGL.gl;
335 		}
336 		if (!chesterGL.gl || !chesterGL.offContext) {
337 			throw "Error initializing graphic context!";
338 		}
339 		chesterGL.webglMode = false;
340 	}
341 	// first resize of the canvas
342 	chesterGL.canvasResized();
343 	
344 	// install touch handler
345 	chesterGL.installMouseHandlers();
346 };

367 /**
368  * init the default shader
369  * @ignore
370  */
371 chesterGL.initDefaultShaders = function () {
372 	var gl = chesterGL.gl;
373 				
374 	chesterGL.initShader("default", function (program) {
375 		program.mvpMatrixUniform = gl.getUniformLocation(program, "uMVPMatrix");
376 		program.attribs = {
377 			'vertexPositionAttribute': gl.getAttribLocation(program, "aVertexPosition"),
378 			'vertexColorAttribute': gl.getAttribLocation(program, "aVertexColor")
379 		}
380 		goog.exportProperty(program, 'mvpMatrixUniform', program.mvpMatrixUniform);
381 		goog.exportProperty(program, 'attribs', program.attribs);
382 	});
383 	
384 	chesterGL.initShader("texture", function (program) {
385 		program.mvpMatrixUniform = gl.getUniformLocation(program, "uMVPMatrix");
386 		program.samplerUniform = gl.getUniformLocation(program, "uSampler");
387 		program.attribs = {
388 			'vertexColorAttribute': gl.getAttribLocation(program, "aVertexColor"),
389 			'textureCoordAttribute': gl.getAttribLocation(program, "aTextureCoord"),
390 			'vertexPositionAttribute': gl.getAttribLocation(program, "aVertexPosition")
391 		};
392 		goog.exportProperty(program, 'mvpMatrixUniform', program.mvpMatrixUniform);
393 		goog.exportProperty(program, 'samplerUniform', program.samplerUniform);
394 		goog.exportProperty(program, 'attribs', program.attribs);
395 	});
396 };

398 /**
399  * init shaders (fetches data - in a sync way)
400  * @param {string} prefix
401  * @param {function(WebGLProgram)} callback
402  * @ignore
403  */
404 chesterGL.initShader = function (prefix, callback) {
405 	var gl = chesterGL.gl;
406 	var fsData = chesterGL.loadShader(prefix, "frag");
407 	var vsData = chesterGL.loadShader(prefix, "vert");
408 
409 	var fs = gl.createShader(gl.FRAGMENT_SHADER);
410 	gl.shaderSource(fs, fsData);
411 	gl.compileShader(fs);
412 	
413 	if (!gl.getShaderParameter(fs, gl.COMPILE_STATUS)) {
414 		console.log("problem compiling fragment shader " + prefix + "(" + gl.getShaderInfoLog(fs) + "):\n" + fsData);
415 		return;
416 	}
417 	
418 	var vs = gl.createShader(gl.VERTEX_SHADER);
419 	gl.shaderSource(vs, vsData);
420 	gl.compileShader(vs);
421 	if (!gl.getShaderParameter(vs, gl.COMPILE_STATUS)) {
422 		console.log("problem compiling vertex shader " + prefix + "(" + gl.getShaderInfoLog(vs) + "):\n" + vsData);
423 		return;
424 	}
425 	
426 	var program = chesterGL.createShader(prefix, fs, vs);
427 	if (callback) callback(program);
428 };

430 /**
431  * loads the shader data
432  * @return {string}
433  * @ignore
434  */
435 chesterGL.loadShader = function (prefix, type) {
436 	var shaderData = "";
437 	$.ajax({
438 		url: "shaders/" + prefix + "." + type,
439 		async: false,
440 		type: 'GET',
441 		success: function (data, textStatus) {
442 			if (textStatus == "success") {
443 				shaderData = data;
444 			} else {
445 				console.log("error getting the shader data");
446 			}
447 		}
448 	});
449 	return shaderData;
450 };

452 /**
453  * actually creates the shader
454  * @param {string} prefix
455  * @param {WebGLShader|null} fragmentData
456  * @param {WebGLShader|null} vertexData
457  * @return {WebGLProgram}
458  * @ignore
459  */
460 chesterGL.createShader = function (prefix, fragmentData, vertexData) {
461 	var gl = chesterGL.gl;
462 	var program = gl.createProgram();
463 	gl.attachShader(program, fragmentData);
464 	gl.attachShader(program, vertexData);
465 	gl.linkProgram(program);
466 	if (!gl.getProgramParameter(program, gl.LINK_STATUS)) {
467 		console.log("problem linking shader");
468 	}
469 	// console.log("creating shader " + prefix);
470 	chesterGL.programs[prefix] = program;
471 	return program;
472 };

474 /**
475  * registers an asset loader (which is basically a function).
476  * By convention, every handler must set the <code>data</code> property of the asset. It should not
477  * modify the status property.
478  * 
479  * @param {string} type the type of the asset
480  * @param {Function} handler the handler that will be called every time an asset is loaded
481  * 
482  * @example
483  * chesterGL.defaultTextureHandler = function (path, data, callback) {
484  *		// create the image
485  *		var imgtype = (/[.]/.exec(path)) ? /[^.]+$/.exec(path) : undefined;
486  *		var img = new Image();
487  *		var rawData = window.base64_encode(data);
488  *		img.onload = function () {
489  *			if (chesterGL.webglMode) {
490  *				img.tex = chesterGL.gl.createTexture();
491  *			}
492  *			var texture = chesterGL.assets['texture'][path];
493  *			texture.data = img;
494  *			var result = true;
495  *			if (chesterGL.webglMode) {
496  *				result = chesterGL.prepareWebGLTexture(img);
497  *			}
498  *			callback && callback(result);
499  *		}
500  *		img.src = "data:image/" + imgtype + ";base64," + rawData;
501  * }
502  */
503 chesterGL.registerAssetHandler = function (type, handler) {
504 	chesterGL.assetsHandlers[type] = handler;
505 };

517 /**
518  * Loads and asset using the registered method to download it. You can register different loaders
519  * (to be called to actually do the request) and asset handlers (to be called after the asset is loaded).
520  * 
521  * @param {string} type the type of asset being loaded, it could be "texture" or "default"
522  * @param {string|Object} assetPath the path for the asset
523  * @param {function(Object)=} cb the callback that will be executed as soon as the asset is loaded
524  */
525 chesterGL.loadAsset = function (type, assetPath, cb) {
526 	var dataType_ = undefined;
527 	if (typeof(assetPath) == 'object') {
528 		dataType_ = assetPath.dataType;
529 		assetPath = assetPath.path;
530 	}
531 
532 	if (!chesterGL.assets[type]) {
533 		chesterGL.assets[type] = {};
534 	}
535 
536 	var assets = chesterGL.assets[type];
537 	if (!assets[assetPath]) {
538 		// not in our records
539 		assets[assetPath] = {
540 			data: null,
541 			status: 'try',
542 			listeners: []
543 		}
544 		cb && assets[assetPath].listeners.push(cb);
545 		chesterGL.loadAsset(type, {path: assetPath, dataType: dataType_});
546 	} else if (assets[assetPath].status == 'loading') {
547 		// created but not yet loaded
548 		cb && assets[assetPath].listeners.push(cb);
549 	} else if (assets[assetPath].status == 'loaded') {
550 		// created and loaded, just call the callback
551 		cb && cb(assets[assetPath].data);
552 	} else if (assets[assetPath].status == 'try') {
553 		assets[assetPath].status = 'loading';
554 		if (chesterGL.assetsLoaders[type])
555 			chesterGL.assetsLoaders[type](type, {url: assetPath, dataType: dataType_});
556 		else
557 			chesterGL.assetsLoaders['default'](type, {url: assetPath, dataType: dataType_});
558 		cb && assets[assetPath].listeners.push(cb);
559 	}
560 };

562 /**
563  * adds a listener/query for when all assets are loaded
564  * 
565  * @param {string} type You can query for all types if you pass "all" as type
566  * @param {function()=} callback The callback to be executed when all assets of that type are loaded
567  */
568 chesterGL.assetsLoaded = function (type, callback) {
569 	var listeners = chesterGL.assetsLoadedListeners[type];
570 	if (!listeners) {
571 		chesterGL.assetsLoadedListeners[type] = [];
572 		listeners = chesterGL.assetsLoadedListeners[type];
573 	}
574 	if (callback) {
575 		listeners.push(callback);
576 	}
577 	var allLoaded = true;
578 	if (type == 'all') {
579 		for (var t in chesterGL.assets) {
580 			var assets = chesterGL.assets[t];
581 			for (var i in assets) {
582 				if (assets[i].status != 'loaded') {
583 					allLoaded = false;
584 					break;
585 				}
586 			}
587 			if (!allLoaded) {
588 				break;
589 			}
590 		}
591 	} else {
592 		var assets = chesterGL.assets[type];
593 		for (var i in assets) {
594 			if (assets[i].status != 'loaded') {
595 				allLoaded = false;
596 				break;
597 			}
598 		}
599 	}
600 	if (allLoaded) {
601 		var l;
602 		while (l = listeners.shift()) { l(); }
603 	}
604 };

619 /**
620  * handles a loaded texture - should only be called on a webGL mode
621  * @param {HTMLImageElement} texture
622  * @return {boolean}
623  * @ignore
624  */
625 chesterGL.prepareWebGLTexture = function (texture) {
626 	var gl = chesterGL.gl;
627 	var result = true;
628 	
629 	try {
630 		var error = 0;
631 		gl.pixelStorei(gl.UNPACK_FLIP_Y_WEBGL, 1);
632 		gl.activeTexture(gl.TEXTURE0);
633 		gl.bindTexture(gl.TEXTURE_2D, texture.tex);
634 		gl.texImage2D(gl.TEXTURE_2D, 0, gl.RGBA, gl.RGBA, gl.UNSIGNED_BYTE, texture);
635 		error = gl.getError();
636 		if (error != 0) {
637 			console.log("gl error " + error);
638 			result = false;
639 		}
640 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MAG_FILTER, gl.LINEAR);
641 		gl.texParameteri(gl.TEXTURE_2D, gl.TEXTURE_MIN_FILTER, gl.LINEAR);
642 		gl.bindTexture(gl.TEXTURE_2D, null);
643 	} catch (e) {
644 		console.log("got some error: " + e);
645 		result = false;
646 	}
647 	return result;
648 };

650 /**
651  * The default texture handler
652  * 
653  * @param {string} path
654  * @param {Object|HTMLImageElement} img
655  * @return {boolean}
656  * @ignore
657  */
658 chesterGL.defaultTextureHandler = function (path, img) {
659 	if (chesterGL.webglMode) {
660 		img.tex = chesterGL.gl.createTexture();
661 	}
662 	var texture = chesterGL.assets['texture'][path];
663 	texture.data = img;
664 	if (chesterGL.webglMode) {
665 		return chesterGL.prepareWebGLTexture(/** @type {HTMLImageElement} */(img));
666 	}
667 	return true;
668 };

670 /**
671  * @param {string} type
672  * @param {Object.<string,string>|null|string} params
673  * @ignore
674  */
675 chesterGL.defaultTextureLoader = function (type, params) {
676 	var img = new Image();
677 	var path = params.url;
678 	img.addEventListener("load", function () {
679 		var texture = chesterGL.assets['texture'][path];
680 		if (chesterGL.assetsHandlers[type](path, img)) {
681 			// call all listeners
682 			texture.status = 'loaded';
683 			var l;
684 			while (l = texture.listeners.shift()) { l(texture.data); }
685 			// test for assets loaded
686 			chesterGL.assetsLoaded(type);
687 			chesterGL.assetsLoaded('all');
688 		} else {
689 			// requeue
690 			texture.status = 'try';
691 			chesterGL.loadAsset(type, path);
692 		}
693 	}, false);
694 	img.src = path;
695 };

697 /**
698  * @param {string} type
699  * @param {Object.<string,string>|null|string} params
700  * @ignore
701  */
702 chesterGL.defaultAssetLoader = function (type, params) {
703 	var path = params.url;
704 	$.ajax({
705 		url: path,
706 		dataType: params.dataType,
707 		success: function (data, textStatus) {
708 			var asset = chesterGL.assets[type][path];
709 			if (textStatus == "success") {
710 				var handler = chesterGL.assetsHandlers[type];
711 				if (!handler) { throw "No handler for asset of type " + type; }
712 				if (handler(path, data)) {
713 					asset.status = 'loaded';
714 					// call all listeners
715 					var l;
716 					while (l = asset.listeners.shift()) { l(asset.data); }
717 					// test for assets loaded
718 					chesterGL.assetsLoaded(type);
719 					chesterGL.assetsLoaded('all');
720 				} else {
721 					// requeue
722 					asset.status = 'try';
723 					chesterGL.loadAsset(type, path);
724 				}
725 			} else {
726 				console.log("Error loading asset " + path);
727 			}
728 		}
729 	});
730 };

732 /**
733  * setups the perspective (2d or 3d)
734  */
735 chesterGL.setupPerspective = function () {
736 	var gl = chesterGL.gl;
737 
738 	// quick bail if we're not on a webgl rendering mode
739 	if (!chesterGL.webglMode) {
740 		return;
741 	}
742 
743 	gl.clearColor(0.0, 0.0, 0.0, 1.0);
744 	// gl.clearDepth(1.0);
745 
746 	// global blending options
747 	gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
748 	gl.enable(gl.BLEND);
749 	// disable depth test
750 	gl.disable(gl.DEPTH_TEST)
751 
752 	var width = gl.viewportWidth;
753 	var height = gl.viewportHeight;
754 	gl.viewport(0, 0, width, height);
755 
756 	chesterGL.pMatrix = goog.vec.Mat4.createFloat32();
757 
758 	if (chesterGL.projection == "2d") {
759 		// 2d projection
760 		console.log("setting up 2d projection (" + width + "," + height + ")");
761 		goog.vec.Mat4.makeOrtho(chesterGL.pMatrix, 0, width, 0, height, -1024, 1024);
762 	} else if (chesterGL.projection == "3d") {
763 		// 3d projection
764 		console.log("setting up 3d projection (" + width + "," + height + ")");
765 		// var f_aspect = (1.7320508075688776 / (width / height));
766 		var zeye   = height / 1.1566;
767 		var matA   = goog.vec.Mat4.makePerspective(goog.vec.Mat4.createFloat32(), 60 * Math.PI / 180.0, width / height, 0.5, 1500.0);
768 		var eye    = [width/2, height/2, zeye];
769 		var center = [width/2, height/2, 0];
770 		var up     = [0, 1, 0];
771 		var matB   = goog.vec.Mat4.makeLookAt(goog.vec.Mat4.createFloat32(), eye, center, up);
772 		goog.vec.Mat4.multMat(matA, matB, chesterGL.pMatrix);
773 	} else {
774 		throw "Invalid projection: " + chesterGL.projection;
775 	}
776 };

788 /**
789  * main draw function, will call the root block
790  * @ignore
791  */
792 chesterGL.drawScene = function () {
793 	var gl = undefined;
794 	if (chesterGL.webglMode) {
795 		gl = chesterGL.gl;
796 		gl.clear(gl.COLOR_BUFFER_BIT | gl.DEPTH_BUFFER_BIT);
797 	} else {
798 		gl = chesterGL.offContext;
799 		gl.setTransform(1, 0, 0, 1, 0, 0);
800 		gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
801 	}
802 	
803 	// start mayhem
804 	if (chesterGL.runningScene) {
805 		chesterGL.runningScene.visit();
806 	}
807 	
808 	if (!chesterGL.webglMode) {
809 		// copy back the off context (if we use one)
810 		if (chesterGL.usesOffscreenBuffer) {
811 			gl.fillRect(0, 0, gl.viewportWidth, gl.viewportHeight);
812 			gl.drawImage(chesterGL.offCanvas, 0, 0);
813 		}
814 	}
815 	// for actions and other stuff
816 	var current = Date.now(); // milliseconds
817 	chesterGL.delta = current - chesterGL.lastTime;
818 	chesterGL.lastTime = current;	
819 };

826 /**
827  * @type {number}
828  * @ignore
829  */
830 chesterGL.elapsed_ = 0;

832  * @type {number}
833  * @ignore
834  */

837  * @type {number}

847 /** @ignore */
848 chesterGL.updateDebugTime = function () {
849 	var now = Date.now();
850 	chesterGL.elapsed_ += chesterGL.delta;
851 	chesterGL.frames_ ++;
852 	if (now - chesterGL.lastDebugSecond_ > 1000) {
853 		var avg = (chesterGL.elapsed_ / chesterGL.frames_);
854 		chesterGL.sumAvg += avg;
855 		chesterGL.sampledAvg ++;
856 		if (chesterGL.debugSpan) {
857 			chesterGL.debugSpan.textContent = avg.toFixed(2);
858 		}
859 		// track how well we're performing - every 5 seconds
860 		if (chesterGL.runningScene && chesterGL.useGoogleAnalytics && chesterGL.sampledAvg > 5) {
861 			_gaq.push([
862 				'_trackEvent',
863 				'chesterGL',
864 				// Let us know if this is WebGL or canvas mode
865 				'renderTime-' + (chesterGL.webglMode) + chesterGL.version,
866 				chesterGL.runningScene.title,
867 				Math.floor(chesterGL.sumAvg/chesterGL.sampledAvg)
868 			]);
869 			chesterGL.sumAvg = chesterGL.sampledAvg = 0;
870 		}
871 		chesterGL.elapsed_ = chesterGL.frames_ = 0;
872 		chesterGL.lastDebugSecond_ = now;
873 	}
874 };

931 /**
932  * Adds a mouse handler: the function will be called for every mouse event on the main canvas
933  * @param {function((Array|Float32Array), chesterGL.mouseEvents)} callback
934  * @example
935  * var stPoint = null;
936  * chesterGL.addMouseHandler(function (pt, type) {
937  * 	if (type == chesterGL.mouseEvents.DOWN) {
938  * 		stPoint = new Float32Array(pt);
939  * 	} else if (type == chesterGL.mouseEvents.MOVE && stPoint) {
940  * 		var tmp = [pt[0] - stPoint[0], pt[1] - stPoint[1], pt[2] - stPoint[2]];
941  * 		tmx.setPosition([tmx.position[0] + tmp[0], tmx.position[1] + tmp[1], tmx.position[2] + tmp[2]]);
942  * 		stPoint.set(pt);
943  * 	} else {
944  * 		stPoint = null;
945  * 	}
946  * });
947  */
948 chesterGL.addMouseHandler = function (callback) {
949 	if (chesterGL.mouseHandlers.indexOf(callback) == -1) {
950 		chesterGL.mouseHandlers.push(callback);
951 	}
952 };

954 /**
955  * Removes a specific mouse handler.
956  * @param {function((Array|Float32Array), chesterGL.mouseEvents)} callback
957  */
958 chesterGL.removeMouseHandler = function (callback) {
959 	var idx = chesterGL.mouseHandlers.indexOf(callback);
960 	if (idx > 0) {

962 	}
