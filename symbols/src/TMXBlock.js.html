 26 (function (window) {
 27 	var ChesterGL = window['ChesterGL'];
 28 	
 29 	/**
 30 	 * unpacks a UInt32 from a string
 31 	 * @ignore
 32 	 */
 33 	function unpackUInt32(buffer, offset) {
 34 		return ((buffer.charCodeAt(offset + 3) & 0xff) << 24 |
 35 				(buffer.charCodeAt(offset + 2) & 0xff) << 16 |
 36 				(buffer.charCodeAt(offset + 1) & 0xff) <<  8 |
 37 				(buffer.charCodeAt(offset + 0) & 0xff));
 38 	}
 39 	
 40 	/**
 41 	 * @ignore
 42 	 * 
 43 	 * @param {string} path
 44 	 * @param {string} data
 45 	 */
 46 	function handleLoadTMX(path, data) {
 47 		console.log("tmx loaded: " + path);
 48 		
 49 		var map = ChesterGL.assets['tmx'][path];
 50 		map.data = data;
 51 		return true;
 52 	}
 53 	
 54 	/**
 55 	 * Creates a new TMXBlock (from a TMX file)
 56 	 * The first line of children are the layers of the tmx map and they're all BlockGroups
 57 	 * 
 58 	 * @constructor
 59 	 * @extends ChesterGL.Block
 60 	 * @param {string} tmxFile
 61 	 * @return ChesterGL.TMXBlock
 62 	 */
 63 	ChesterGL.TMXBlock = function (tmxFile) {
 64 		// first, get the map from the map list
 65 		var map = ChesterGL.TMXBlock.maps[tmxFile];
 66 		if (!map) {
 67 			throw "Invalid map - make sure you call loadTMX first";
 68 		}
 69 		ChesterGL.Block.call(this, null, ChesterGL.Block.TYPE['TMXBLOCK']);
 70 		// create the map from the metadata
 71 		for (var i=0; i < map['layers'].length; i++) {
 72 			var layer = map['layers'][i];
 73 			// this is while block group is not supported on canvas fallback
 74 			var l = (ChesterGL.webglMode ? new ChesterGL.BlockGroup(map['texture'], layer['blocks'].length) : new ChesterGL.Block());
 75 			for (var n=0; n < layer['blocks'].length; n++) {
 76 				var block = layer['blocks'][n];
 77 				var b = undefined;
 78 				if (ChesterGL.webglMode) {
 79 					b = l.createBlock(block['frame']);
 80 				} else {
 81 					b = new ChesterGL.Block(block['frame']);
 82 					b.setTexture(map['texture']);
 83 				}
 84 				b.moveTo(block['position']);
 85 				l.addChild(b);
 86 			}
 87 			this.addChild(l);
 88 		}
 89 	};
 90

 92 	ChesterGL.TMXBlock.prototype = Object.create(ChesterGL.Block.prototype);

121 	ChesterGL.TMXBlock.prototype.spacing = 0;

131 	ChesterGL.TMXBlock.maps = {};

133 	/**
134 	 * Will load a TMX file, parse it when loaded and add the metadata to create
135 	 * the TMX block later
136 	 * 
137 	 * @param {string} path
138 	 */
139 	ChesterGL.TMXBlock.loadTMX = function(path) {
140 		ChesterGL.loadAsset('tmx', {path:path, dataType: 'xml'}, function (data) {
141 			var tmx = {};
142 			
143 			var map = $(data).find("map");
144 			var tileset = map.find("tileset").first();
145 			var orientation = map.attr("orientation");
146 			if (tileset) {
147 				tmx['tileSize'] = vec2.create([
148 					parseInt(tileset.attr("tilewidth"), 10),
149 					parseInt(tileset.attr("tileheight"), 10)
150 				]);
151 				tmx['mapTileSize'] = vec2.create([
152 					parseInt(map.attr("tilewidth"), 10),
153 					parseInt(map.attr("tileheight"), 10)
154 				]);
155 				if (tileset.attr("spacing")) {
156 					tmx['spacing'] = parseInt(tileset.attr("spacing"), 10);
157 				}
158 				if (tileset.attr("margin")) {
159 					tmx['margin'] = parseInt(tileset.attr("margin"), 10);
160 				}
161 				
162 				// find the image for the tileset
163 				var image = tileset.find("image").first();
164 				var imageSize = vec2.create([
165 					parseInt(image.attr("width"), 10),
166 					parseInt(image.attr("height"), 10)
167 				]);
168 				tmx['texture'] = image.attr('source');
169 				ChesterGL.loadAsset('texture', tmx['texture']);
170 				
171 				// parse the layers
172 				tmx['layers'] = [];
173 				map.find("layer").each(function (i, layer) {
174 					var blockLayer = {};
175 					blockLayer['blocks'] = [];
176 					
177 					var layerSize = vec2.create([
178 						parseInt($(layer).attr("width"), 10),
179 						parseInt($(layer).attr("height"), 10)
180 					]);
181 					var data = $(layer).find("data").first();
182 					if (data) {
183 						if (data.attr("encoding") != "base64" || data.attr("compression")) {
184 							throw "Invalid TMX Data";
185 						}
186 						var str = data.text().trim();
187 						var decodedData = base64.decode(str);
188 						// fun begins here
189 						var offset = 0;
190 						for (var row = 0; row < layerSize[1]; row++) {
191 							for (var col = 0; col < layerSize[0]; col++) {
192 								var gid = unpackUInt32(decodedData, offset) - 1;
193 								var b = {};
194 								var margin = tmx['margin'] || 0;
195 								var spacing = tmx['spacing'] || 0;
196 								var tileSize = tmx['tileSize'];
197 								var mapTileSize = tmx['mapTileSize'];
198 								
199 								var max_x = parseInt((imageSize[0] - margin * 2 + spacing) / (tileSize[0] + spacing), 10);
200 								var frame = quat4.create([
201 									// assume gid == 1
202 									(gid % max_x) * (tileSize[0] + spacing) + margin,
203 									(imageSize[1] - tileSize[1] - margin - spacing) - parseInt(gid / max_x, 10) * (tileSize[1] + spacing) + margin,
204 									tileSize[0],
205 									tileSize[1]
206 								]);
207 								// console.log("gid " + col + "," + row + ": " + gid + "; frame: " + frame.l + "," + frame.t);
208 								b['frame'] = frame;
209 								var bx, by;
210 								if (orientation == "orthogonal") {
211 									bx = col * mapTileSize[0]                     + tileSize[0]/2;
212 									by = (layerSize[1] - row - 1) * mapTileSize[1] + tileSize[1]/2;
213 								} else if (orientation == "isometric") {
214 									bx = mapTileSize[0]/2 * (layerSize[0] + col - row - 1)       + tileSize[0]/2;
215 									by = mapTileSize[1]/2 * ((layerSize[1] * 2 - col - row) - 2) + tileSize[1]/2;
216 								} else {
217 									throw "Invalid orientation";
218 								}
219 								b['position'] = [bx, by, 0];
220 								blockLayer['blocks'].push(b);
221 								offset += 4;
222 							}
223 						}
224 					} else {
225 						throw "No data for layer!"
226 					}
227 					tmx['layers'].push(blockLayer);
228 				}); // each layer
229 			} // if tileset
230 			ChesterGL.TMXBlock.maps[path] = tmx;
231 		});
232 	};
233 		
234 	// just register a dummy handler
235 	ChesterGL.registerAssetHandler('tmx', handleLoadTMX);
236 	
237 	// export symbols
238 	ChesterGL['TMXBlock'] = ChesterGL.TMXBlock;
239 	// class methods
240 	ChesterGL.exportProperty(ChesterGL.TMXBlock, 'loadTMX', ChesterGL.TMXBlock.loadTMX);
241 })(window);
