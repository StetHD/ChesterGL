 29 	/**
 30 	 * creates a new block. Pass the rect if you want to set the frame at
 31 	 * creation time.
 32 	 * 
 33 	 * @constructor
 34 	 * @param {Object.<string,number>|string=} rect
 35 	 * @param {number=} type
 36 	 * @param {ChesterGL.Block=} parent
 37 	 * @return ChesterGL.Block
 38 	 */
 39 	ChesterGL.Block = function (rect, type, parent) {
 40 		this.type = type || ChesterGL.Block.TYPE.STANDALONE;
 41 		if (parent) {
 42 			this.parent = parent;
 43 		}
 44 		
 45 		this.children = [];
 46 		this.program = ChesterGL.Block.PROGRAM.DEFAULT;
 47 		
 48 		if (rect) {
 49 			if (typeof(rect) === 'string') {
 50 				var f = ChesterGL.BlockFrames.getFrame(rect);
 51 				this.setTexture(f.texture);
 52 				this.setFrame(f.frame);
 53 			} else {
 54 				this.setFrame(rect);
 55 			}
 56 		}
 57 		// set default color
 58 		if (this.type == ChesterGL.Block.TYPE.STANDALONE) {
 59 			this.setColor([1, 1, 1, 1]);
 60 		}
 61 		
 62 		if (ChesterGL.webglMode && this.type == ChesterGL.Block.TYPE.STANDALONE && (!parent || parent.type != ChesterGL.Block.TYPE.BLOCKGROUP)) {
 63 			var gl = ChesterGL.gl;
 64 			// just a single buffer for all data (a 4x"quad")
 65 			this.glBuffer = gl.createBuffer();
 66 			this.glBufferData = new Float32Array(ChesterGL.Block.BUFFER_SIZE);
 67 		}
 68 		
 69 		// always create the mvMatrix
 70 		this.mvMatrix = mat4.create();
 71 		mat4.identity(this.mvMatrix);
 72 	}

290 	ChesterGL.Block.prototype.setTexture = function (texturePath) {
291 		this.texture = texturePath;
292 		// force program to texture program
293 		this.program = ChesterGL.Block.PROGRAM.TEXTURE;
294 		var block = this;
295 		ChesterGL.loadAsset("texture", texturePath, function (t) {
296 			// set the default frame for all our blocks (if it's not set)
297 			if (!block.contentSize) {
298 				block.setContentSize([t.width, t.height]);
299 			}
300 			if (!block.frame) {
301 				block.setFrame([0, 0, t.width, t.height]);
302 			}
303 		});
304 	}

306 	/**
307 	 * move the block to a specific location.
308 	 * You can pass a time in seconds to make this an action
309 	 * 
310 	 * @param {vec3|Array} vec
311 	 * @param {number=} time The time it should take to move to that position. Don't pass time to make it instantly
312 	 */
313 	ChesterGL.Block.prototype.moveTo = function (vec, time) {
314 		if (time) {
315 			var a = new ChesterGL.MoveAction(this, time, vec);
316 			ChesterGL.ActionManager.scheduleAction(a);
317 		} else {
318 			this.position = vec3.create(vec);
319 			this.isTransformDirty = true;
320 		}
321 	}
322 	
323 	/**
324 	 * move the block relatively
325 	 * @param {vec3|Array} vec
326 	 * @param {number=} time The time it should take to move to that position. Don't pass time to make it instantly
327 	 */
328 	ChesterGL.Block.prototype.moveBy = function (vec, time) {
329 		vec3.add(this.position, vec);
330 		this.isTransformDirty = true;
331 	}

364 	ChesterGL.Block.prototype.transform = function () {
365 		var gl = ChesterGL.gl;
366 		var transformDirty = (this.isTransformDirty || (this.parent && this.parent.isTransformDirty));
367 		if (transformDirty) {
368 			mat4.identity(this.mvMatrix);
369 			mat4.translate(this.mvMatrix, this.position);
370 			mat4.rotate(this.mvMatrix, this.rotation, [0, 0, 1]);
371 			mat4.scale(this.mvMatrix, [this.scale, this.scale, 1]);
372 			// concat with parent's transform
373 			var ptransform = (this.parent ? this.parent.mvMatrix : null);
374 			if (ptransform) {
375 				mat4.multiply(ptransform, this.mvMatrix, this.mvMatrix);
376 			}
377 		}
378 		
379 		// bail out if we're a block group
380 		if (this.type == ChesterGL.Block.TYPE.BLOCKGROUP) {
381 			return;
382 		}
383 		
384 		var bufferData = this.glBufferData;
385 		var inBlockGroup = this.parent && this.parent.type == ChesterGL.Block.TYPE.BLOCKGROUP;
386 		
387 		if (ChesterGL.webglMode) {
388 			if (!inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
389 				gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
390 			}
391 			if (this.isFrameDirty || (inBlockGroup && this.isTransformDirty)) {
392 				// NOTE
393 				// the tex coords and the frame coords need to match. Otherwise you get a distorted image
394 				var offset = 9;
395 				var hw = this.contentSize[0] * 0.5, hh = this.contentSize[1] * 0.5;
396 				var _idx = this.baseBufferIndex * ChesterGL.Block.BUFFER_SIZE;
397 				var z = this.position[2];
398 
399 				// NOTE
400 				// this is going to be slow :P
401 				// this is using the matrix to transform the vertex data
402 				if (inBlockGroup) {
403 					var tr = [ hw,  hh, 0],
404 						tl = [-hw,  hh, 0],
405 						br = [ hw, -hh, 0],
406 						bl = [-hw, -hh, 0];
407 					mat4.multiplyVec3(this.mvMatrix, tr);
408 					mat4.multiplyVec3(this.mvMatrix, tl);
409 					mat4.multiplyVec3(this.mvMatrix, bl);
410 					mat4.multiplyVec3(this.mvMatrix, br);
411 
412 					bufferData[_idx           ] =  bl[0]; bufferData[_idx + 1           ] =  bl[1]; bufferData[_idx + 2           ] = z;
413 					bufferData[_idx +   offset] =  tl[0]; bufferData[_idx + 1 +   offset] =  tl[1]; bufferData[_idx + 2 +   offset] = z;
414 					bufferData[_idx + 2*offset] =  br[0]; bufferData[_idx + 1 + 2*offset] =  br[1]; bufferData[_idx + 2 + 2*offset] = z;
415 					bufferData[_idx + 3*offset] =  tr[0]; bufferData[_idx + 1 + 3*offset] =  tr[1]; bufferData[_idx + 2 + 3*offset] = z;
416 				} else {
417 					bufferData[_idx           ] = -hw; bufferData[_idx + 1           ] = -hh; bufferData[_idx + 2           ] = 0;
418 					bufferData[_idx +   offset] = -hw; bufferData[_idx + 1 +   offset] =  hh; bufferData[_idx + 2 +   offset] = 0;
419 					bufferData[_idx + 2*offset] =  hw; bufferData[_idx + 1 + 2*offset] = -hh; bufferData[_idx + 2 + 2*offset] = 0;
420 					bufferData[_idx + 3*offset] =  hw; bufferData[_idx + 1 + 3*offset] =  hh; bufferData[_idx + 2 + 3*offset] = 0;
421 				}
422 
423 				if (this.program == ChesterGL.Block.PROGRAM.TEXTURE) {
424 					var tex = ChesterGL.getAsset("texture", this.texture);
425 					var texW = tex.width,
426 						texH = tex.height;
427 					var l = this.frame[0] / texW,
428 						b = this.frame[1] / texH,
429 						w = this.frame[2] / texW,
430 						h = this.frame[3] / texH;
431 					_idx += 3;
432 					bufferData[_idx           ] = l  ; bufferData[_idx+1           ] = b;
433 					bufferData[_idx +   offset] = l  ; bufferData[_idx+1 +   offset] = b+h;
434 					bufferData[_idx + 2*offset] = l+w; bufferData[_idx+1 + 2*offset] = b;
435 					bufferData[_idx + 3*offset] = l+w; bufferData[_idx+1 + 3*offset] = b+h;
436 				}
437 			}
438 			if (this.isColorDirty) {
439 				_idx = 5 + this.baseBufferIndex * ChesterGL.Block.BUFFER_SIZE;
440 				var color = this.color;
441 				var opacity = this.opacity;
442 				for (var i=0; i < 4; i++) {
443 					bufferData[_idx     + offset*i] = color[0] * opacity;
444 					bufferData[_idx + 1 + offset*i] = color[1] * opacity;
445 					bufferData[_idx + 2 + offset*i] = color[2] * opacity;
446 					bufferData[_idx + 3 + offset*i] = color[3] * opacity;
447 				}
448 			}
449 			if (ChesterGL.webglMode && !inBlockGroup && (this.isFrameDirty || this.isColorDirty)) {
450 				gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
451 			}
452 		}
453 	}

455 	ChesterGL.Block.prototype.visit = function () {
456 		if (this.update) {
457 			this.update();
458 		}
459 		if (!this.visible) {
460 			return;
461 		}
462 		this.transform();
463 		
464 		var children = this.children;
465 		var len = children.length;
466 		for (var i=0; i < len; i++) {
467 			children[i].visit();
468 		}
469 		
470 		// render this block if not in a block group
471 		if (!this.parent || this.parent.type != ChesterGL.Block.TYPE.BLOCKGROUP) {
472 			this.render();
473 		}
474 		// reset our dirty markers
475 		this.isFrameDirty = this.isColorDirty = this.isTransformDirty = false;
476 	}

478 	/**
479 	 * render (only will work for non-blockgroup blocks)
480 	 */
481 	ChesterGL.Block.prototype.render = function () {
482 		if (this.type == ChesterGL.Block.TYPE.BLOCKGROUP) {
483 			throw "Cannot call render on a BlockGroup block!";
484 		}
485 		// empty render for scene blocks
486 		if (this.type == ChesterGL.Block.TYPE.SCENE) {
487 			return;
488 		}
489 		
490 		if (ChesterGL.webglMode) {
491 			var gl = ChesterGL.gl;
492 			// select current shader
493 			var program = ChesterGL.selectProgram(ChesterGL.Block.PROGRAM_NAME[this.program]);
494 
495 			gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
496 			var texOff = 3 * 4,
497 				colorOff = texOff + 2 * 4,
498 				stride = ChesterGL.Block.QUAD_SIZE;
499 			
500 			gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
501 			gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, colorOff);
502 
503 			if (this.program == ChesterGL.Block.PROGRAM.DEFAULT) {
504 				// no extra attributes for the shader
505 			} else if (this.program == ChesterGL.Block.PROGRAM.TEXTURE) {
506 				var texture = ChesterGL.getAsset('texture', this.texture);
507 
508 				// pass the texture attributes
509 				gl.vertexAttribPointer(program.attribs['textureCoordAttribute'], 2, gl.FLOAT, false, stride, texOff);
510 
511 				gl.activeTexture(gl.TEXTURE0);
512 				gl.bindTexture(gl.TEXTURE_2D, texture.tex);
513 				gl.uniform1i(program.samplerUniform, 0);				
514 			}
515 
516 			// set the matrix uniform (actually, only the model view matrix)
517 			gl.uniformMatrix4fv(program.mvMatrixUniform, false, this.mvMatrix);
518 			gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4);
519 		} else {
520 			var gl = ChesterGL.offContext;
521 			// canvas drawing api - we only draw textures
522 			if (this.program == ChesterGL.Block.PROGRAM.TEXTURE) {
523 				var m = this.mvMatrix;
524 				var texture = ChesterGL.getAsset('texture', this.texture);
525 				gl.globalAlpha = this.opacity;
526 				gl.setTransform(m[0], m[1], m[4], m[5], m[12], gl.viewportHeight - m[13]);
527 				var w = this.contentSize[0], h = this.contentSize[1];
528 				var frame = this.frame;
529 				gl.drawImage(texture, frame[0], texture.height - (frame[1] + h), frame[2], frame[3], -w/2, -h/2, w, h);
530 			}
531 		}
532 	}
