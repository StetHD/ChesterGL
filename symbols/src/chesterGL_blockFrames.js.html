 26 goog.provide("chesterGL.BlockFrames");
 27 
 28 goog.require("goog.vec.Vec4");
 29 goog.require("chesterGL.Block");
 30 
 31 /**
 32  * The namespace for BlockFrame related functions
 33  * 
 34  * @const
 35  */
 36 chesterGL.BlockFrames = {};
 37 
 38 /**
 39  * @type {Object.<string,chesterGL.Block.frameType>}
 40  */
 41 chesterGL.BlockFrames.frames = {};
 42 
 43 /**
 44  * loads the json data (callback for the ajax call)
 45  * 
 46  * @param {Object} data
 47  */
 48 chesterGL.BlockFrames.loadJSON = function (data) {
 49 	// first, get the meta data
 50 	if (data['meta'] && data['meta']['version'] == '1.0') {
 51 		var texName = data['meta']['image'];
 52 		chesterGL.loadAsset('texture', texName, function (img) {
 53 			var imgHeight = img.height;
 54 			var frames = data['frames'];
 55 			for (var frameName in frames) {
 56 				var f = frames[frameName];
 57 				/** @type {chesterGL.Block.frameType} */ var realFrame = {frame: {}, texture: ""};
 58 				realFrame.frame = goog.vec.Vec4.createFloat32FromValues(
 59 					f['frame']['x'],
 60 					imgHeight - (f['frame']['y'] + f['frame']['h']),
 61 					f['frame']['w'],
 62 					f['frame']['h']
 63 				);
 64 				realFrame.texture = texName;
 65 				chesterGL.BlockFrames.frames[frameName] = realFrame;
 66 			}
 67 		});
 68 	} else {
 69 		throw "Unkown json data";
 70 	}
 71 };
 72 
 73 /**
 74  * @param {string} path
 75  * @param {string|Object} data
 76  * @return {boolean}
 77  * @ignore
 78  */
 79 chesterGL.BlockFrames.framesLoadHandler = function (path, data) {
 80 	var frameset = chesterGL.assets['frameset'][path];
 81 	frameset.data = data;
 82 	return true;
 83 };
 84 
 85 /**
 86  * Returns a named frame.
 87  *
 88  * @param {string} frameName
 89  * @return {chesterGL.Block.frameType}
 90  */
 91 chesterGL.BlockFrames.getFrame = function (frameName) {
 92 	return chesterGL.BlockFrames.frames[frameName];
 93 };
 94 
 95 /**
 96  * @param {string} path The path for the json file (TexturePacker format)
 97  * @param {function()=} callback The callback to be called after everything is ready
 98  */
 99 chesterGL.BlockFrames.loadFrames = function (path, callback) {
100 	console.log("loadFrames: will fetch " + path);
101 	chesterGL.loadAsset("frameset", {path: path, dataType: 'json'}, function (data) {
102 		chesterGL.BlockFrames.loadJSON(data);
103 	});
104 };
105 
106 chesterGL.registerAssetHandler('frameset', chesterGL.BlockFrames.framesLoadHandler);
107 
108 // export symbols
109 goog.exportSymbol('chesterGL.BlockFrames', chesterGL.BlockFrames);
110 // static methods
111 goog.exportProperty(chesterGL.BlockFrames, 'getFrame', chesterGL.BlockFrames.getFrame);
112 goog.exportProperty(chesterGL.BlockFrames, 'loadFrames', chesterGL.BlockFrames.loadFrames);
