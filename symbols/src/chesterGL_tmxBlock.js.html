 33 /**
 34  * Creates a new TMXBlock (from a TMX file)
 35  * The first line of children are the layers of the tmx map and they're all BlockGroups
 36  * 
 37  * @constructor
 38  * @param {string} tmxFile
 39  * @extends chesterGL.Block
 40  */
 41 chesterGL.TMXBlock = function (tmxFile) {
 42 	// first, get the map from the map list
 43 	var map = chesterGL.TMXBlock.maps[tmxFile];
 44 	if (!map) {
 45 		throw "Invalid map - make sure you call loadTMX first";
 46 	}
 47 	chesterGL.Block.call(this, null, chesterGL.Block.TYPE['TMXBLOCK']);
 48 	// create the map from the metadata
 49 	for (var i=0; i < map['layers'].length; i++) {
 50 		var layer = map['layers'][i];
 51 		// this is while block group is not supported on canvas fallback
 52 		var l = (chesterGL.webglMode ? new chesterGL.BlockGroup(null, layer['blocks'].length) : new chesterGL.Block());
 53 		var tileset = null;
 54 		for (var n=0; n < layer['blocks'].length; n++) {
 55 			// meta block
 56 			var mblock = layer['blocks'][n];
 57 			// set the texture for the block group
 58 			if (!tileset) {
 59 				tileset = chesterGL.TMXBlock.findTilesetForGid(map['tilesets'], mblock['gid']);
 60 				l.setTexture(tileset['texture']);
 61 			}
 62 			var b = undefined;
 63 			if (chesterGL.webglMode) {
 64 				b = l.createBlock(mblock['frame']);
 65 			} else {
 66 				b = new chesterGL.Block(mblock['frame']);
 67 				b.setTexture(tileset['texture']);
 68 			}
 69 			b.setPosition(mblock['position']);
 70 			l.addChild(b);
 71 		}
 72 		this.addChild(l);
 73 	}
 74 };
 75 goog.inherits(chesterGL.TMXBlock, chesterGL.Block);

122 /**
123  * unpacks a UInt32 from a string
124  * @param {Array} buffer
125  * @param {number} offset
126  * @ignore
127  */
128 chesterGL.TMXBlock.unpackUInt32 = function (buffer, offset) {
129 	return (buffer[offset + 3] << 24) |
130 		   (buffer[offset + 2] << 16) |
131 		   (buffer[offset + 1] <<  8) |
132 		    buffer[offset + 0] >>> 0;
133 };

135 /**
136  * @param {string} path
137  * @param {string} data
138  * @ignore
139  */
140 chesterGL.TMXBlock.handleLoadTMX = function (path, data) {
141 	var map = chesterGL.assets['tmx'][path];
142 	map.data = data;
143 	return true;
144 };
145 
146 /**
147  * finds the right texture for the given gid
148  * @param {Array.<Object>} tilesets
149  * @param {number} gid
150  * @returns {Object}
151  */
152 chesterGL.TMXBlock.findTilesetForGid = function (tilesets, gid) {
153 	var tex = tilesets[0];
154 	for (var i=1; i < tilesets.length; i++) {
155 		var ts = tilesets[i];
156 		if (gid >= ts['firstgid']) {
157 			tex = ts;
158 		}
159 	}
160 	return tex;
161 };

163 /**
164  * Will load a TMX file, parse it when loaded and add the metadata to create
165  * the TMX block later
166  * 
167  * @param {string} path
168  */
169 chesterGL.TMXBlock.loadTMX = function (path) {
170 	chesterGL.loadAsset('tmx', {path:path, dataType: 'xml'}, function (data) {
171 		var tmx = {};
172 		
173 		var map = $(data).find("map");
174 		var orientation = map.attr("orientation");
175 		tmx['tilesets'] = [];
176 		map.find("tileset").each(function (i, xtileset) {
177 			var ts = $(xtileset);
178 			if (ts.attr('name') != "obstruction") {
179 				var tileset = {};
180 				tileset['tileSize'] = new goog.math.Size(
181 					parseInt(ts.attr("tilewidth"), 10),
182 					parseInt(ts.attr("tileheight"), 10)
183 				);
184 				if (ts.attr("spacing")) {
185 					tileset['spacing'] = parseInt(ts.attr("spacing"), 10);
186 				}
187 				if (ts.attr("margin")) {
188 					tileset['margin'] = parseInt(ts.attr("margin"), 10);
189 				}
190 				var image = ts.find("image").first();
191 				tileset['imgSize'] = new goog.math.Size(
192 					parseInt(image.attr("width"), 10),
193 					parseInt(image.attr("height"), 10)
194 				);
195 				tileset['texture'] = image.attr('source');
196 				tileset['firstgid'] = parseInt(ts.attr('firstgid'), 10);
197 				tmx['tilesets'].push(tileset);
198 			}
199 		});
200 		tmx['mapTileSize'] = new goog.math.Size(
201 			parseInt(map.attr("tilewidth"), 10),
202 			parseInt(map.attr("tileheight"), 10)
203 		);
204 		// parse the layers
205 		tmx['layers'] = [];
206 		map.find("layer").each(function (i, layer) {
207 			if ($(layer).attr("visible") != "0") {
208 				var blockLayer = {};
209 				blockLayer['blocks'] = [];
210 
211 				var layerSize = new goog.math.Size(
212 					parseInt($(layer).attr("width"), 10),
213 					parseInt($(layer).attr("height"), 10)
214 				);
215 				var data = $(layer).find("data").first();
216 				if (data) {
217 					if (data.attr("encoding") != "base64" || data.attr("compression")) {
218 						throw "Invalid TMX Data";
219 					}
220 					var str = data.text().trim();
221 					var decodedData = goog.crypt.base64.decodeStringToByteArray(str);
222 					// fun begins here
223 					var offset = 0;
224 					var tileset = null;
225 					for (var row = 0; row < layerSize.height; row++) {
226 						for (var col = 0; col < layerSize.width; col++) {
227 							var gid = chesterGL.TMXBlock.unpackUInt32(decodedData, offset);
228 							if (gid == 0) {
229 								offset += 4;
230 								continue;
231 							}
232 							if (!tileset) {
233 								tileset = chesterGL.TMXBlock.findTilesetForGid(tmx['tilesets'], gid);
234 							}
235 							var b = {};
236 							b['gid'] = gid;
237 							var margin = tileset['margin'] || 0;
238 							var spacing = tileset['spacing'] || 0;
239 							var tileSize = tileset['tileSize'];
240 							var imageSize = tileset['imgSize'];
241 							var mapTileSize = tmx['mapTileSize'];
242 
243 							var max_x = parseInt((imageSize.width - margin * 2 + spacing) / (tileSize.width + spacing), 10);
244 							gid = gid - tileset['firstgid'];
245 							var frame = goog.vec.Vec4.createFloat32FromValues(
246 								(gid % max_x) * (tileSize.width + spacing) + margin,
247 								(imageSize.height - tileSize.height - margin - spacing) - parseInt(gid / max_x, 10) * (tileSize.height + spacing) + margin,
248 								tileSize.width,
249 								tileSize.height
250 							);
251 							// console.log("gid " + col + "," + row + ": " + gid + "; frame: " + frame.l + "," + frame.t);
252 							b['frame'] = frame;
253 							var bx, by;
254 							if (orientation == "orthogonal") {
255 								bx = col * mapTileSize.width                           + tileSize.width/2;
256 								by = (layerSize.height - row - 1) * mapTileSize.height + tileSize.height/2;
257 							} else if (orientation == "isometric") {
258 								bx = mapTileSize.width/2  * (layerSize.width + col - row - 1)        + tileSize.width/2;
259 								by = mapTileSize.height/2 * ((layerSize.height * 2 - col - row) - 2) + tileSize.height/2;
260 							} else {
261 								throw "Invalid orientation";
262 							}
263 							b['position'] = [bx, by, 0];
264 							blockLayer['blocks'].push(b);
265 							offset += 4;
266 						}
267 					}
268 				} else {
269 					throw "No data for layer!"
270 				}
271 				tmx['layers'].push(blockLayer);
272 			}
273 		}); // each layer
274 		chesterGL.TMXBlock.maps[path] = tmx;
275 	});
276 };
