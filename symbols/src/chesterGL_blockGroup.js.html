 30 /**
 31  * creates a new block group
 32  * 
 33  * @constructor
 34  * @param {string} texture An optional texture that will be shared with all children
 35  * @param {number=} noChildren The optional initial number of maxChidlren. Defaults to 10
 36  * @extends {chesterGL.Block}
 37  */
 38 chesterGL.BlockGroup = function (texture, noChildren) {
 39 	if (!chesterGL.webglMode) {
 40 		throw "BlockGroup only works on WebGL mode";
 41 	}
 42 	// call "super"
 43 	chesterGL.Block.call(this, null, chesterGL.Block.TYPE['BLOCKGROUP']);
 44 	if (texture) {
 45 		this.texture = texture;
 46 		this.program = chesterGL.Block.PROGRAM['TEXTURE'];
 47 	} else {
 48 		this.program = chesterGL.Block.PROGRAM['DEFAULT'];
 49 	}
 50 	this.maxChildren = noChildren || 10;
 51 	
 52 	// create the buffers for the children
 53 	this.createBuffers();
 54 }
 55 goog.inherits(chesterGL.BlockGroup, chesterGL.Block);
 56 
 57 /**
 58  * The max number of children this blockgroup can hold (this is here for the buffer data)
 59  * @type {number}
 60  */
 61 chesterGL.BlockGroup.prototype.maxChildren = 0;
 62 
 63 /**
 64  * @type {boolean}
 65  */
 66 chesterGL.BlockGroup.prototype.isChildDirty = false;
 67 		
 68 /**
 69  * @type {?WebGLBuffer}
 70  */
 71 chesterGL.BlockGroup.prototype.indexBuffer = null;
 72 
 73 /**
 74  * @type {?Uint16Array}
 75  */
 76 chesterGL.BlockGroup.prototype.indexBufferData = null;
 77 
 78 /**
 79  * Creates the buffers for the current maxChildren size
 80  * @ignore
 81  */
 82 chesterGL.BlockGroup.prototype.createBuffers = function () {
 83 	var gl = chesterGL.gl;
 84 	this.glBuffer        = gl.createBuffer();
 85 	this.glBufferData    = new Float32Array(chesterGL.Block.QUAD_SIZE * this.maxChildren);
 86 	this.indexBuffer     = gl.createBuffer();
 87 	this.indexBufferData = new Uint16Array(6 * this.maxChildren);
 88 }
 89 
 90 /**
 91  * creates a block that can be added to this block group
 92  * @param {quat4} rect
 93  */
 94 chesterGL.BlockGroup.prototype.createBlock = function (rect) {
 95 	var b = new chesterGL.Block(rect, chesterGL.Block.TYPE['STANDALONE'], this);
 96 	if (this.texture) {
 97 		b.setTexture(this.texture);
 98 	}
 99 	return b;
100 }
101 
102 /**
103  * adds a child
104  * @param {chesterGL.Block} block
105  */
106 chesterGL.BlockGroup.prototype.addChild = function (block) {
107 	if (this.children.length >= this.maxChildren) {
108 		throw "Error: too many children - Make the initial size of the BlockGroup larger"
109 	}
110 	if (block.parent != this) {
111 		throw "Invalid child: can only add children created with BlockGroup.create";
112 	}
113 	if (!this.texture) {
114 		this.texture = block.texture;
115 	} else {
116 		if (this.texture != block.texture) {
117 			throw "Invalid child: only can add child with the same texture";
118 		}
119 	}
120 	this.children.push(block);
121 	
122 	var length = this.children.length;
123 
124 	// just point the buffer data on the child and set the baseIndex
125 	block.baseBufferIndex = length-1;
126 	block.glBufferData    = this.glBufferData;
127 	this.isChildDirty = true;
128 }
129 
130 /**
131  * when the block list changes, the indices array must be recreated
132  * @param {number} startIdx
133  */
134 chesterGL.BlockGroup.prototype.recreateIndices = function (startIdx) {
135 	var lastIdx = (this.indexBufferData[startIdx*6 - 1] || -1) + 1;
136 	var total = Math.max(this.children.length, 1);
137 	for (var i = startIdx; i < total; i ++) {
138 		var idx = i*6;
139 		this.indexBufferData[idx + 0] = lastIdx    ; this.indexBufferData[idx + 1] = lastIdx + 1; this.indexBufferData[idx + 2] = lastIdx + 2;
140 		this.indexBufferData[idx + 3] = lastIdx + 2; this.indexBufferData[idx + 4] = lastIdx + 1; this.indexBufferData[idx + 5] = lastIdx + 3;
141 		lastIdx += 4;
142 	}
143 	var gl = chesterGL.gl;
144 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.indexBuffer);
145 	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, this.indexBufferData, gl.STATIC_DRAW);
146 }

156 /**
157  * where the fun begins
158  */
159 chesterGL.BlockGroup.prototype.visit = function () {
160 	if (this.update) {
161 		this.update(chesterGL.delta);
162 	}
163 	if (!this.visible) {
164 		return;
165 	}
166 	this.transform();
167 	
168 	// this should be more like cocos2d: preserve the z-ordering
169 	var children = this.children;
170 	var len = children.length;
171 	for (var i=0; i < len; i++) {
172 		children[i].visit();
173 	}
174 	
175 	// re-bind the glBuffer (might have changed in the children visit)
176 	var gl = chesterGL.gl;
177 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
178 	gl.bufferData(gl.ARRAY_BUFFER, this.glBufferData, gl.STATIC_DRAW);
179 
180 	if (this.isChildDirty) {
181 		this.recreateIndices(0);
182 		this.isChildDirty = false;
183 	}
184 	// render this block group
185 	this.render();
186 	
187 	// reset our dirty markers
188 	this.isFrameDirty = this.isColorDirty = this.isTransformDirty = false;
189 }
190 
191 /**
192  * actually render the block group
193  */
194 chesterGL.BlockGroup.prototype.render = function () {
195 	var gl = chesterGL.gl;
196 	
197 	// select current shader
198 	var program = chesterGL.selectProgram(chesterGL.Block.PROGRAM_NAME[this.program]);
199 	var totalChildren = this.children.length;
200 	var texOff = 3 * 4,
201 		colorOff = texOff + 2 * 4,
202 		stride = chesterGL.Block.QUAD_SIZE;
203 	
204 	gl.bindBuffer(gl.ARRAY_BUFFER, this.glBuffer);
205 	
206 	gl.vertexAttribPointer(program.attribs['vertexPositionAttribute'], 3, gl.FLOAT, false, stride, 0);
207 	if (this.program == chesterGL.Block.PROGRAM['DEFAULT']) {
208 		// no extra attributes for the shader
209 	} else if (this.program == chesterGL.Block.PROGRAM['TEXTURE']) {
210 		var texture = chesterGL.getAsset('texture', this.texture);
211 		
212 		// pass the texture attributes
213 		gl.vertexAttribPointer(program.attribs['textureCoordAttribute'], 2, gl.FLOAT, false, stride, texOff);
214 		
215 		gl.activeTexture(gl.TEXTURE0);
216 		gl.bindTexture(gl.TEXTURE_2D, texture.tex);
217 		gl.uniform1i(program.samplerUniform, 0);
218 	}
219 	gl.vertexAttribPointer(program.attribs['vertexColorAttribute'], 4, gl.FLOAT, false, stride, colorOff);
220 	
221 	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.indexBuffer);
222 	
223 	// set the matrix uniform (the multiplied model view projection matrix)
224 	mat4.multiply(chesterGL.pMatrix, this.mvMatrix, this.mvpMatrix);
225 	gl.uniformMatrix4fv(program.mvpMatrixUniform, false, this.mvpMatrix);
226 	// gl.drawArrays(gl.TRIANGLE_STRIP, 0, 4 * this.children.length);
227 	gl.drawElements(gl.TRIANGLES, totalChildren * 6, gl.UNSIGNED_SHORT, 0);
228 }
